88 /*FUNCTION**********************************************************************
89  *
90  * Function Name : SystemCoreClockUpdate
91  * Description : This function must be called whenever the core clock is changed
92  * during program execution. It evaluates the clock register settings and calculates
93  * the current core clock.
94  *
95  * Implements : SystemCoreClockUpdate_Activity
96  *END**************************************************************************/
97 void SystemCoreClockUpdate(void)
98 {
99  uint32_t SCGOUTClock = 0U; /* Variable to store output clock frequency of the SCG module */
100  uint32_t regValue; /* Temporary variable */
101  uint32_t divider;
102  bool validSystemClockSource = true;
103 
104  divider = ((SCG->CSR & SCG_CSR_DIVCORE_MASK) >> SCG_CSR_DIVCORE_SHIFT) + 1U;
105 
106  switch ((SCG->CSR & SCG_CSR_SCS_MASK) >> SCG_CSR_SCS_SHIFT)
107  {
108  case 0x1:
109  /* System OSC */
110  SCGOUTClock = CPU_XTAL_CLK_HZ;
111  break;
112  case 0x2:
113  /* Slow IRC */
114  regValue = (SCG->SIRCCFG & SCG_SIRCCFG_RANGE_MASK) >> SCG_SIRCCFG_RANGE_SHIFT;
115  if (regValue != 0UL)
116  {
117  SCGOUTClock = FEATURE_SCG_SIRC_HIGH_RANGE_FREQ;
118  }
119  else
120  {
121  validSystemClockSource = false;
122  }
123  break;
124  case 0x3:
125  /* Fast IRC */
126  regValue = (SCG->FIRCCFG & SCG_FIRCCFG_RANGE_MASK) >> SCG_FIRCCFG_RANGE_SHIFT;
127  if (regValue == 0x0UL)
128  {
129  SCGOUTClock = FEATURE_SCG_FIRC_FREQ0;
130  }
131  else
132  {
133  validSystemClockSource = false;
134  }
135  break;
136  default:
137  validSystemClockSource = false;
138  break;
139  }
140 
141  if (validSystemClockSource == true)
142  {
143  SystemCoreClock = (SCGOUTClock / divider);
144  }
145 }

147 /*FUNCTION**********************************************************************
148  *
149  * Function Name : SystemSoftwareReset
150  * Description : This function is used to initiate a system reset
151  *
152  * Implements : SystemSoftwareReset_Activity
153  *END**************************************************************************/
154 void SystemSoftwareReset(void)
155 {
156  uint32_t regValue;
157 
158  /* Read Application Interrupt and Reset Control Register */
159  regValue = S32_SCB->AIRCR;
160 
161  /* Clear register key */
162  regValue &= ~( S32_SCB_AIRCR_VECTKEY_MASK);
163 
164  /* Configure System reset request bit and Register Key */
165  regValue |= S32_SCB_AIRCR_VECTKEY(FEATURE_SCB_VECTKEY);
166  regValue |= S32_SCB_AIRCR_SYSRESETREQ(0x1u);
167 
168  /* Write computed register value */
169  S32_SCB->AIRCR = regValue;
170 }
