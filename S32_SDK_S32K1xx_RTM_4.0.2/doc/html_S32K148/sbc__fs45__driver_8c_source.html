210  * Function Name : FS45_IsRegisterSecured
211  * Description : Returns true if register SPI transfer is secured by security bits.
212  *
213  *END**************************************************************************/
214 static inline bool FS45_IsRegisterSecured(uint8_t address)
215 {
216  uint8_t i; /* Counter. */
217 
218  static const uint8_t m_securedRegisters[] = {
219  FS45_M_MODE_ADDR,
220  FS45_M_REG_MODE_ADDR,
221  FS45_FS_INIT_FS1B_TIMING_ADDR,
222  FS45_FS_BIST_ADDR,
223  FS45_FS_INIT_SUPERVISOR_ADDR,
224  FS45_FS_INIT_FAULT_ADDR,
225  FS45_FS_INIT_FSSM_ADDR,
226  FS45_FS_INIT_SF_IMPACT_ADDR,
227  FS45_FS_WD_WINDOW_ADDR,
228  FS45_FS_SF_OUTPUT_REQUEST_ADDR,
229  FS45_FS_INIT_WD_CNT_ADDR,
230  FS45_FS_INIT_VCORE_OVUV_IMPACT_ADDR,
231  FS45_FS_INIT_VCCA_OVUV_IMPACT_ADDR,
232  FS45_FS_INIT_VAUX_OVUV_IMPACT_ADDR
233  };
234 
235  for (i = 0; i < FS45_SECURE_REGS_COUNT; i++)
236  {
237  if (address == m_securedRegisters[i])
238  {
239  return true;
240  }
241  }
242  return false;
243 }

245 /*FUNCTION**********************************************************************
246  *
247  * Function Name : FS45_TransferData
248  * Description : Performs SPI transfer of the txData. Received frame is saved into
249  * rxData structure or ignored if rxData is NULL.
250  *
251  *END**************************************************************************/
252 static status_t FS45_TransferData(const fs45_tx_data_t* txData, fs45_rx_data_t* rxData)
253 {
254  status_t status; /* Status variable. */
255  uint8_t txFrame[FS45_SPI_FRAME_SIZE_BYTES] = {0}; /* SPI Tx frame. */
256  uint8_t rxFrame[FS45_SPI_FRAME_SIZE_BYTES] = {0}; /* SPI Rx frame. */
257 
258  /* Create frame to send. */
259  FS45_CreateSendFrame(txData, txFrame);
260 
261  status = LPSPI_DRV_MasterTransferBlocking(g_spiInstance, txFrame, rxFrame, 2U, g_spiTimeout);
262 
263  if (status != STATUS_SUCCESS)
264  {
265  return status;
266  }
267 
268  if (rxData != NULL)
269  {
270  /* Secured fail-safe registers and RELEASE_FSxB register contain extended status. */
271  if (((txData->isSecured) && (FS45_IS_REG_FAILSAFE(txData->registerAddress) != 0U)) ||
272  (txData->registerAddress == FS45_FS_RELEASE_FSXB_ADDR))
273  {
274  rxData->deviceStatusEx = rxFrame[FS45_SPI_BIT_7_0] & FS45_SPI_DEVICE_STATUS_EX_MASK;
275  rxData->readData = rxFrame[FS45_SPI_BIT_7_0] & FS45_SECURE_DATA_READ_MASK;
276  }
277  else
278  {
279  rxData->deviceStatusEx = 0x00U;
280  rxData->readData = rxFrame[FS45_SPI_BIT_7_0];
281  }
282  rxData->deviceStatus = rxFrame[FS45_SPI_BIT_15_8];
283  }
284  return status;
285 }

287 /*FUNCTION**********************************************************************
288  *
289  * Function Name : FS45_CreateSendFrame
290  * Description : Creates a raw frame for SPI transfer.
291  *
292  *END**************************************************************************/
293 static void FS45_CreateSendFrame(const fs45_tx_data_t* txData, uint8_t* txFrame)
294 {
295  fs45_parity_t parity; /* Tx frame parity result. */
296 
297  txFrame[FS45_SPI_BIT_7_0] = 0x00U;
298  txFrame[FS45_SPI_BIT_15_8] = 0x00U;
299 
300  /* Sets address of the register. */
301  txFrame[FS45_SPI_BIT_15_8] = FS45_BO_SETVAL_EXT(txFrame[FS45_SPI_BIT_15_8],
302  txData->registerAddress, FS45_REG_ADDR_MASK, FS45_REG_ADDR_SHIFT);
303 
304  /* Check if register is r or w */
305  switch (txData->commandType)
306  {
307  case fs45RegRead:
308  /* Creates read command. */
309  txFrame[FS45_SPI_BIT_15_8] = FS45_BO_SETVAL(txFrame[FS45_SPI_BIT_15_8],
310  FS45_SPI_R_BIT_VALUE, FS45_SPI_RW_BIT_MASK);
311  /* No data to be sent. */
312  txFrame[FS45_SPI_BIT_7_0] = 0x00U;
313  break;
314 
315  case fs45RegWrite:
316  /* Creates write command. */
317  txFrame[FS45_SPI_BIT_15_8] |= FS45_BO_SETVAL(txFrame[FS45_SPI_BIT_15_8],
318  FS45_SPI_W_BIT_VALUE, FS45_SPI_RW_BIT_MASK);
319  /* Sets data. */
320  txFrame[FS45_SPI_BIT_7_0] = txData->writeData;
321  if (txData->isSecured)
322  {
323  FS45_ComputeSecurityBits(&(txFrame[FS45_SPI_BIT_7_0]));
324  }
325  /* Sets parity bit. */
326  parity = FS45_ComputeParity((((uint16_t)(txFrame[FS45_SPI_BIT_15_8]) << 8U)) + (uint16_t)(txFrame[FS45_SPI_BIT_7_0]));
327  if (parity == fs45ParityEven)
328  {
329  txFrame[FS45_SPI_BIT_15_8] = FS45_BO_SETVAL(txFrame[FS45_SPI_BIT_15_8], 1U, 1U);
330  }
331  break;
332  default:
333  /* Do nothing. */
334  break;
335  }
336 }

338 /*FUNCTION**********************************************************************
339  *
340  * Function Name : FS45_WriteRegisters
341  * Description : Writes set of configuration registers values.
342  *
343  *END**************************************************************************/
344 static status_t FS45_WriteRegisters(const fs45_reg_config_value_t* registers, uint8_t numOfItems)
345 {
346  status_t firstStatus; /* Status variable. */
347  status_t secondStatus; /* Status variable. */
348  fs45_rx_data_t response; /* Response to the command. */
349  uint8_t i; /* Counter. */
350  const fs45_reg_config_value_t* reg; /* Pointer to actually selected register configuration. */
351  uint8_t writeValue; /* Register write value. */
352 
353  if ((registers == NULL) || (numOfItems == 0U))
354  {
355  return STATUS_SUCCESS;
356  }
357 
358  for (i = 0; i < numOfItems; i++)
359  {
360  reg = &(registers[i]);
361  /* Secured registers have the write value shifted and 4 lowest bits
362  * are used for the Secured bits. */
363  writeValue = FS45_IsRegisterSecured(reg->address) ?
364  (uint8_t)(reg->value << FS45_SECURE_WRITE_SHIFT) : reg->value;
365  firstStatus = FS45_WriteRegister(reg->address, writeValue, NULL);
366 
367  /* In case of write to the fail-safe register, add mandatory delay between
368  * consequent transfers (3.5 us according to the specification, 1 ms added). */
369  if (FS45_IS_REG_FAILSAFE(reg->address) != 0U)
370  {
371  OSIF_TimeDelay(1);
372  }
373 
374  /* Check written register value. */
375  secondStatus = FS45_ReadRegister(reg->address, &response);
376 
377  /* Read mask should be used for read value as some registers
378  * contains status bits, that are not writable (e.g. INIT_VREG - BAT_FAIL bit) */
379  if ((firstStatus != STATUS_SUCCESS) || (secondStatus != STATUS_SUCCESS) || ((response.readData & reg->readMask) != (reg->value & reg->readMask)))
380  {
381  return STATUS_ERROR;
382  }
383  }
384  return STATUS_SUCCESS;
385 }

387 /*FUNCTION**********************************************************************
388  *
389  * Function Name : FS45_ComputeSecurityBits
390  * Description : Computes security bits from 4 MSbits and write it to 4 LSbits.
391  *
392  *END**************************************************************************/
393 static void FS45_ComputeSecurityBits(uint8_t* data)
394 {
395  /* Set 4 LSbits to 0. */
396  *data &= (uint8_t)0xF0U;
397 
398  *data |= (((*data & (1U << 6U)) != 0U) ? 1U : 0U); /* Secure 0 = Bit6 */
399  *data |= (((*data & (1U << 7U)) != 0U) ? 2U : 0U); /* Secure 1 = Bit7 */
400  *data |= (((*data & (1U << 4U)) != 0U) ? 0U : 4U); /* Secure 2 = NOT(Bit4) */
401  *data |= (((*data & (1U << 5U)) != 0U) ? 0U : 8U); /* Secure 3 = NOT(Bit5) */
402 }

404 /*FUNCTION**********************************************************************
405  *
406  * Function Name : FS45_ComputeParity
407  * Description : This function calculates parity value of passed data array.
408  * Returns 1 if the parity is even, 0 otherwise.
409  *
410  *END**************************************************************************/
411 static fs45_parity_t FS45_ComputeParity(uint16_t data)
412 {
413  uint16_t x = data; /* Data from which the parity will be calculated. */
414 
415  x ^= x >> 8U;
416  x ^= x >> 4U;
417  x ^= x >> 2U;
418  x ^= x >> 1U;
419 
420  return (((~x) & 1U) != 0U) ? fs45ParityEven : fs45ParityOdd;
421 }

452 static uint8_t FS45_WD_ComputeAnswer(uint8_t token)
453 {
454  uint16_t mr = token; /* Monitoring result. */
455 
456  /* Compute challenger watchdog answer. */
457  mr *= 4U;
458  mr += 6U;
459  mr -= 4U;
460  mr = ~mr;
461  mr /= 4U;
462 
463  return (uint8_t)mr;

479 {

481  uint32_t statusInt = 0U; /* Status variable. */
482  fs45_rx_data_t response; /* Response to the command. */
483  fs45_current_mode_t sbcMode; /* Actual SBC mode. */
484  fs45_prev_mode_t prevMode; /* Previous SBC mode. */
485  fs45_reg_config_value_t initMainRegs[] =
486  {
487  { FS45_M_INIT_VREG_ADDR, 0x00U, 0xF6U },
488  { FS45_M_INIT_WU1_ADDR, 0x00U, 0xFFU },
489  { FS45_M_INIT_WU2_ADDR, 0x00U, 0xF7U },
490  { FS45_M_INIT_INH_INT_ADDR, 0x00U, 0x1FU }
491  };
492  fs45_reg_config_value_t initFailSafeRegs[] =
493  {
494  { FS45_FS_INIT_FS1B_TIMING_ADDR, 0x00U, 0x0FU },
495  { FS45_FS_INIT_SUPERVISOR_ADDR, 0x00U, 0x0FU },
496  { FS45_FS_INIT_FAULT_ADDR, 0x00U, 0x0FU },
497  { FS45_FS_INIT_FSSM_ADDR, 0x00U, 0x0FU },
498  { FS45_FS_INIT_SF_IMPACT_ADDR, 0x00U, 0x0FU },
499  { FS45_FS_INIT_WD_CNT_ADDR, 0x00U, 0x0FU },
500  { FS45_FS_INIT_VCORE_OVUV_IMPACT_ADDR, 0x00U, 0x0FU },
501  { FS45_FS_INIT_VCCA_OVUV_IMPACT_ADDR, 0x00U, 0x0FU },
502  { FS45_FS_INIT_VAUX_OVUV_IMPACT_ADDR, 0x00U, 0x0FU }
503  };
504  uint8_t initIntValue;
505 
506  DEV_ASSERT(drvConfig != NULL);
507 
508  g_spiInstance = lpspiInstance;
509  g_spiTimeout = lpspiTimeoutMs;
510 
511  /* Check if FS1B is implemented. */
512  status = FS45_ReadRegister(FS45_FS_DEVICE_ID_FS_ADDR, &response);
513  if (status != STATUS_SUCCESS)
514  {
515  return status;
516  }
517 
518  /* Store to global variable. */
519  g_fs1bEnabled = (((response.readData & FS45_R_FS_FS1_MASK) >> FS45_R_FS_FS1_SHIFT) != 0U) ? true : false;
520 
521  /* Check built-in self-test. */
522  status = FS45_CheckLbistAbistOk();
523  if (status != STATUS_SUCCESS)
524  {
525  return status;
526  }
527 
528  /* Check if FS1B is at low (if implemented). */
529  if (g_fs1bEnabled == true)
530  {
531  status = FS45_CheckFS1B();
532  if (status != STATUS_SUCCESS)
533  {
534  return status;
535  }
536  }
537 
538  /* Determine if it is necessary to re-initialize main register set. */
539  status = FS45_GetMode(&sbcMode, &prevMode);
540  if (status != STATUS_SUCCESS)
541  {
542  return status;
543  }
544 
545  /* MSM does not have to be initialized in case of transition
546  * LPOFF -> NORMAL. */
547  if ((sbcMode == fs45ModeInit) && (prevMode != fs45PrevModeLPOFF))
548  {
549  /* Initialize main registers:
550  * - INIT_VREG
551  * - INIT_WU1
552  * - INIT_WU2
553  * - INIT_INH_INT
554  */
555  initMainRegs[0].value = drvConfig->vregReg.iccaLim |

558  drvConfig->vregReg.vcanOvMon |

562  drvConfig->wu1Reg.wuIo2 |
563  drvConfig->wu1Reg.wuIo3 |
564  drvConfig->wu1Reg.wuIo4;
565  initMainRegs[2].value = drvConfig->wu2Reg.wuIo5 |
566  drvConfig->wu2Reg.canDisCfg |
567  drvConfig->wu2Reg.canWuTo |
568  drvConfig->wu2Reg.linJ2602Dis |
569  drvConfig->wu2Reg.linSr;
570  initMainRegs[3].value = drvConfig->inhIntReg.inh5 |
571  drvConfig->inhIntReg.inh4 |
572  drvConfig->inhIntReg.inh3 |
573  drvConfig->inhIntReg.inh2 |
574  drvConfig->inhIntReg.inh0;
575 
576  status = FS45_WriteRegisters(initMainRegs, 4U);
577  if (status != STATUS_SUCCESS)
578  {
579  return status;
580  }
581  }
582 
583  /* FSSM can be initialized only in INIT_FS. */
584  if (sbcMode == fs45ModeInit)
585  {
586  /* Initialize FSSM (fail-safe registers):
587  * - INIT_FS1B_TIMING
588  * - INIT_SUPERVISOR
589  * - INIT_FAULT
590  * - INIT_FSSM
591  * - INIT_SF_IMPACT
592  * - INIT_WD_CNT
593  * - INIT_VCORE_OVUV_IMPACT
594  * - INIT_VCCA_OVUV_IMPACT
595  * - INIT_VAUX_OVUV_IMPACT
596  */
597  initFailSafeRegs[0].value = drvConfig->fs1bTimingReg.fs1bTime;
598  initFailSafeRegs[1].value = drvConfig->supervisorReg.vcore5d |
599  drvConfig->supervisorReg.vcca5d |
600  drvConfig->supervisorReg.vaux5d |
601  drvConfig->supervisorReg.fs1bTimeRange;
602  initFailSafeRegs[2].value = drvConfig->faultReg.fltErrFs |
603  drvConfig->faultReg.fs1bCanImpact |
604  drvConfig->faultReg.fltErrImp;
605  initFailSafeRegs[3].value = drvConfig->fssmReg.io45Fs |
606  drvConfig->fssmReg.io23Fs |
607  drvConfig->fssmReg.ps |
608  drvConfig->fssmReg.rstbDuration;
609  initFailSafeRegs[4].value = drvConfig->sfImpactReg.tdlyTdur |
610  drvConfig->sfImpactReg.dis8s |
611  drvConfig->sfImpactReg.wdImpact;
612  initFailSafeRegs[5].value = drvConfig->wdCntReg.wdCntErr |
613  drvConfig->wdCntReg.wdCntRfr;
614  initFailSafeRegs[6].value = drvConfig->vcoreOvuvImpactReg.vcoreFsOv |
615  drvConfig->vcoreOvuvImpactReg.vcoreFsUv;
616  initFailSafeRegs[7].value = drvConfig->vccaOvuvImpactReg.vccaFsOv |
617  drvConfig->vccaOvuvImpactReg.vccaFsUv;
618  initFailSafeRegs[8].value = drvConfig->vauxOvuvImpactReg.vauxFsOv |
619  drvConfig->vauxOvuvImpactReg.vauxFsUv;
620 
621  /* Check if device is in debug mode. */
622  status = FS45_ReadRegister(FS45_M_HW_CONFIG_ADDR, &response);
623  if (status != STATUS_SUCCESS)
624  {
625  return status;
626  }
627 
628  if((response.readData & FS45_R_M_DBG_HW_MASK) == FS45_R_M_DBG_HW_DEBUG)
629  {
630  /* Change INIT_FAULT register mask - FS1B_CAN_IMP bit is 0 by default in debug mode. */
631  initFailSafeRegs[2].readMask = 0x0BU;
632  }
633 
634  status = FS45_WriteRegisters(initFailSafeRegs, 9U);
635  if (status != STATUS_SUCCESS)
636  {
637  return status;
638  }
639 
640  /* INIT_MAIN -> NORMAL_MODE. */
641  initIntValue = drvConfig->intReg.duration |
642  drvConfig->intReg.inhLin |
643  drvConfig->intReg.inhAll |
644  drvConfig->intReg.inhVsns |
645  drvConfig->intReg.inhVpre |
646  drvConfig->intReg.inhVcore |
647  drvConfig->intReg.inhVother |
648  drvConfig->intReg.inhCan;
649 
650  status = FS45_WriteRegister(FS45_M_INIT_INT_ADDR, initIntValue, NULL);
651  if (status != STATUS_SUCCESS)
652  {
653  return status;
654  }
655  }
656 
657  /* Read Diag. registers to clear all bits. */
658  statusInt = (uint32_t)FS45_ReadRegister(FS45_M_DIAG_VPRE_ADDR, &response);
659  statusInt|= (uint32_t)FS45_ReadRegister(FS45_M_DIAG_VCORE_ADDR, &response);
660  statusInt|= (uint32_t)FS45_ReadRegister(FS45_M_DIAG_VCCA_ADDR, &response);
661  statusInt|= (uint32_t)FS45_ReadRegister(FS45_M_DIAG_VAUX_ADDR, &response);
662  statusInt|= (uint32_t)FS45_ReadRegister(FS45_M_DIAG_VSUP_VCAN_ADDR, &response);
663  if (statusInt != (uint32_t)STATUS_SUCCESS)
664  {
665  return STATUS_ERROR;
666  }
667 
668  /* Initialize the rest of registers.
669  * If nonInitRegs is NULL, or nonInitRegsCount is 0, this step will be skipped. */
670  status = FS45_WriteRegisters(drvConfig->nonInitRegs, drvConfig->nonInitRegsCount);
671  if (status != STATUS_SUCCESS)
672  {
673  return status;
674  }
675 
676  /* INIT_FS -> NORMAL_WD */
677  status = FS45_WD_Refresh();
678  if (status != STATUS_SUCCESS)
679  {
680  return status;
681  }
682 
683  return STATUS_SUCCESS;
684 }

885 /*FUNCTION**********************************************************************
886  *
887  * Function Name : FS45_SetRegulatorState
888  * Description : Sets state (enable/disable) of the selected voltage regulator.
889  *
890  * Implements : FS45_SetRegulatorState_Activity
891  *
892  *END**************************************************************************/
893 status_t FS45_SetRegulatorState(fs45_reg_mode_t vreg, bool enable)
894 {
895  status_t status; /* Status variable. */
896  uint8_t writeValue = 0U; /* Register write value. */
897  uint8_t address = 0U; /* Register address. */
898  uint8_t readMask = 0U; /* Register read mask. */
899  uint8_t writeMask = 0U; /* Register write mask. */
900  uint8_t writeShift = 0U; /* Register write shift. */
901  fs45_rx_data_t response; /* Received data. */
902  uint8_t secureWriteShift = FS45_SECURE_WRITE_SHIFT; /* Shift for reg. write data. */
903  bool readActualValue = true; /* If true, register actual value is read before write operation. */
904  uint8_t enableValue = enable ? 1U : 0U; /* Enable/disable regulator output. */
905 
906  switch (vreg)
907  {
908  case fs45VCan:
909  address = FS45_M_REG_MODE_ADDR;
910  readMask = FS45_R_M_VCAN_EN_MASK;
911  writeMask = FS45_W_M_VCAN_EN_MASK;
912  writeShift = FS45_W_M_VCAN_EN_SHIFT;
913  break;
914 
915  case fs45Aux:
916  address = FS45_M_REG_MODE_ADDR;
917  readMask = FS45_R_M_VAUX_EN_MASK;
918  writeMask = FS45_W_M_VAUX_EN_MASK;
919  writeShift = FS45_W_M_VAUX_EN_SHIFT;
920  break;
921 
922  case fs45Vcca:
923  address = FS45_M_REG_MODE_ADDR;
924  readMask = FS45_R_M_VCCA_EN_MASK;
925  writeMask = FS45_W_M_VCCA_EN_MASK;
926  writeShift = FS45_W_M_VCCA_EN_SHIFT;
927  break;
928 
929  case fs45Vcore:
930  address = FS45_M_REG_MODE_ADDR;
931  readMask = FS45_R_M_VCORE_EN_MASK;
932  writeMask = FS45_W_M_VCORE_EN_MASK;
933  writeShift = FS45_W_M_VCORE_EN_SHIFT;
934  break;
935 
936  case fs45Vkam:
937  address = FS45_M_MODE_ADDR;
938  readMask = (uint8_t)FS45_RW_M_VKAM_EN_MASK;
939  writeMask = (uint8_t)FS45_RW_M_VKAM_EN_MASK;
940  writeShift = FS45_RW_M_VKAM_EN_SHIFT;
941  /* Value is not shifted for M_MODE register. */
942  secureWriteShift = 0;
943  /* Vkam bit is the only non-volatile value that can be set by M_MODE register.
944  * It has no sense to read content of this register before write. */
945  readActualValue = false;
946  break;
947 
948  default:
949  /* Not valid parameter. */
950  break;
951  }
952 
953  if (readActualValue)
954  {
955  /* Read actual register value to preserve value of other bits. */
956  status = FS45_ReadRegister(address, &response);
957  if (status != STATUS_SUCCESS)
958  {
959  return status;
960  }
961  /* Compose write value. */
962  writeValue = FS45_BO_SETVAL((uint8_t)(response.readData << secureWriteShift),
963  (uint8_t)(enableValue << writeShift), writeMask);
964  }
965  else
966  {
967  /* Write just the enable/disable bit (M_MODE register only). */
968  writeValue = (uint8_t)(enableValue << writeShift);
969  }
970 
971  /* It is not necessary to check status. If someting goes wrong the next check catches the error. */
972  (void)FS45_WriteRegister(address, writeValue, NULL);
973 
974  /* Check if the value was accepted. */
975  status = FS45_ReadRegister(address, &response);
976  if ((status != STATUS_SUCCESS) || (((response.readData & readMask) << secureWriteShift) != (writeValue & writeMask)))
977  {
978  return STATUS_ERROR;
979  }
980 
981  return STATUS_SUCCESS;
982 }
