173  *
174  * Function Name : ENET_DRV_SetTxBufferDescriptors
175  * Description : Configures the software transmit buffer descriptors.
176  *
177  *END**************************************************************************/
178 static void ENET_DRV_SetTxBufferDescriptors(const enet_config_t *config,
179  enet_buffer_descriptor_t *txBdStartAlign,
180  uint32_t txRingSize,
181  uint8_t ring)
182 {
183  uint32_t count;
184  enet_buffer_descriptor_t *curBuffDescrip = txBdStartAlign;
185 
186 #ifdef FEATURE_ENET_TX_MIN_RING_SIZE
187  DEV_ASSERT(txRingSize >= FEATURE_ENET_TX_MIN_RING_SIZE);
188 #endif
189 
190  for (count = 0; count < txRingSize; count++)
191  {
192  curBuffDescrip->control = ENET_BUFFDESCR_TX_TRANSMITCRC_MASK;
193 
194 #if FEATURE_ENET_HAS_ENHANCED_BD
195 #if (FEATURE_ENET_RING_COUNT > 1U)
196  curBuffDescrip->enh1 = (uint32_t)ring << (uint32_t)ENET_TX_ENH1_FTYPE_SHIFT;
197 #else
198  (void)ring;
199  curBuffDescrip->enh1 = 0UL;
200 #endif /* (FEATURE_ENET_RING_COUNT > 1U) */
201  curBuffDescrip->enh2 = 0UL;
202  curBuffDescrip->enh3 = 0UL;
203  if ((config->interrupts & ENET_TX_INTERRUPTS_MASK) != 0UL)
204  {
205  curBuffDescrip->enh1 |= ENET_TX_ENH1_INT_MASK;
206  }
207  if ((config->txAccelerConfig & ENET_TACC_IPCHK_MASK) != 0U)
208  {
209  curBuffDescrip->enh1 |= ENET_TX_ENH1_IINS_MASK;
210  }
211  if ((config->txAccelerConfig & ENET_TACC_PROCHK_MASK) != 0U)
212  {
213  curBuffDescrip->enh1 |= ENET_TX_ENH1_PINS_MASK;
214  }
215  curBuffDescrip->reserved1 = 0UL;
216  curBuffDescrip->reserved2 = 0UL;
217 #else
218  (void)ring;
219  (void)config;
220 #endif /* FEATURE_ENET_HAS_ENHANCED_BD */
221 
222  /* Sets the last buffer descriptor with the wrap flag. */
223  if (count == (txRingSize - 1U))
224  {
225  curBuffDescrip->control |= ENET_BUFFDESCR_TX_WRAP_MASK;
226  }
227 
228  /* Increase the index. */
229  curBuffDescrip++;
230  }
231 }

284  *
285  * Function Name : ENET_DRV_ComputeCRC32
286  * Description : Computes the CRC32 of a given MAC address.
287  *
288  *END**************************************************************************/
289 static uint32_t ENET_DRV_ComputeCRC32(const uint8_t *mac)
290 {
291  uint32_t crc = 0xFFFFFFFFUL;
292  uint8_t i, j;
293 
294  for (i = 0; i < 6U; i++)
295  {
296  crc = crc ^ mac[i];
297  for (j = 0; j < 8U; j++)
298  {
299  if ((crc & 0x1U) != 0U)
300  {
301  crc = (crc >> 1U) ^ 0xEDB88320U;
302  }
303  else
304  {
305  crc = (crc >> 1U);
306  }
307  }
308  }
309 
310  return crc;
311 }

746 /*FUNCTION**********************************************************************
747  *
748  * Function Name : ENET_DRV_GetTransmitStatus
749  * Description : Checks if the transmission of a buffer is complete
750  *
751  * This function checks if the transmission of the given buffer is complete.
752  *
753  * Implements : ENET_DRV_GetTransmitStatus_Activity
754  *END**************************************************************************/
755 status_t ENET_DRV_GetTransmitStatus(uint8_t instance,
756  uint8_t queue,
757  const enet_buffer_t * buff,
758  enet_tx_enh_info_t * info)
759 {
760  const enet_buffer_descriptor_t *bd;
761  status_t status = STATUS_ENET_BUFF_NOT_FOUND;
762  bool finished = false;
763 
764  DEV_ASSERT(instance < ENET_INSTANCE_COUNT);
765  DEV_ASSERT(g_enetState[instance] != NULL);
766  DEV_ASSERT(queue < g_enetState[instance]->ringCount);
767  DEV_ASSERT(buff != NULL);
768 #if (FEATURE_ENET_HAS_ENHANCED_BD == 0)
769  DEV_ASSERT(info == NULL);
770 #endif /* !FEATURE_ENET_HAS_ENHANCED_BD */
771 
772  bd = g_enetState[instance]->txBdBase[queue];
773 
774  /* Search for the buffer descriptor associated with the given buffer. */
775  while (!finished)
776  {
777  if (bd->buffer == buff->data)
778  {
779  /* Check if the buffer descriptor is still in use. */
780  if ((bd->control & ENET_BUFFDESCR_TX_READY_MASK) != 0U)
781  {
782  status = STATUS_BUSY;
783  }
784  else
785  {
786  status = STATUS_SUCCESS;
787 #if FEATURE_ENET_HAS_ENHANCED_BD
788  if (info != NULL)
789  {
790  info->errMask = bd->enh1 & ENET_TX_ENH_ERR_MASK;
791  info->timestamp = bd->timestamp;
792  }
793 #endif /* FEATURE_ENET_HAS_ENHANCED_BD */
794  }
795 
796  break;
797  }
798 
799  /* This was the last buffer in the ring. */
800  if ((bd->control & ENET_BUFFDESCR_TX_WRAP_MASK) != 0U)
801  {
802  finished = true;
803  }
804 
805  bd++;
806  }
807 
808  return status;
809 }
