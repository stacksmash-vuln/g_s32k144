155 /*FUNCTION**********************************************************************
156  *
157  * Function Name : init
158  * Description : Initializes the specified PHY device.
159  *
160  *END**************************************************************************/
161 static status_t init(uint8_t phy)
162 {
163  status_t status;
164  uint16_t basicCtrl;
165  uint16_t mask;
166  phy_speed_t speed = g_phyConfig[phy].speed;
167 
168  if (speed != PHY_SPEED_AUTONEG)
169  {
170  /* set fixed speed */
171  mask = (uint16_t) (BASIC_CONTROL_AUTONEG_FLAG | BASIC_CONTROL_SPEED_MSB_FLAG | BASIC_CONTROL_SPEED_LSB_FLAG);
172  if (speed == PHY_SPEED_100_MBPS)
173  {
174  basicCtrl = BASIC_CONTROL_SPEED_LSB_FLAG;
175  }
176  else if (speed == PHY_SPEED_1_GBPS)
177  {
178  basicCtrl = BASIC_CONTROL_SPEED_MSB_FLAG;
179  }
180  else
181  {
182  basicCtrl = 0;
183  }
184  status = PHY_RMW(phy, BASIC_CONTROL_ADDR, basicCtrl, mask);
185  if (status != STATUS_SUCCESS)
186  {
187  return status;
188  }
189  }
190 
191  status = checkCapabilities(phy);
192  if (status != STATUS_SUCCESS)
193  {
194  return status;
195  }
196 
197  if (g_phyConfig[phy].role != PHY_ROLE_AUTO)
198  {
199  status = PHY_SetRole(phy, g_phyConfig[phy].role);
200  if (status != STATUS_SUCCESS)
201  {
202  return status;
203  }
204  }
205 
206  g_phyConfig[phy].privateData = (g_phyConfig[phy].privateData | PRIVATE_ACTIVE); /* assert active bit */
207  if (g_phyConfig[phy].manualDelay == false)
208  {
209  /* delays unspecified for generic device. Store defaults */
210  g_phyConfig[phy].delay.miiToMdiMax = 0xFFFFFFFFU;
211  g_phyConfig[phy].delay.miiToMdiMin = 0;
212  g_phyConfig[phy].delay.mdiToMiiMax = 0xFFFFFFFFU;
213  g_phyConfig[phy].delay.mdiToMiiMin = 0;
214  }
215  return status;
216 }

218 /*FUNCTION**********************************************************************
219  *
220  * Function Name : reset
221  * Description : Resets the specified PHY device.
222  *
223  *END**************************************************************************/
224 static status_t reset(uint8_t phy)
225 {
226  status_t status;
227  uint16_t basicCtrl;
228  uint32_t startTime, currentTime;
229 
230  status = PHY_Write(phy, BASIC_CONTROL_ADDR, BASIC_CONTROL_RESET_FLAG);
231  if (status == STATUS_SUCCESS)
232  {
233  startTime = OSIF_GetMilliseconds();
234  do
235  {
236  status = PHY_Read(phy, BASIC_CONTROL_ADDR, &basicCtrl);
237  if (((basicCtrl & BASIC_CONTROL_RESET_FLAG) == 0U) || (status != STATUS_SUCCESS))
238  {
239  break;
240  }
241  currentTime = OSIF_GetMilliseconds();
242  }
243  while (currentTime <= (uint32_t) (startTime + T_RST_MAX));
244  }
245 
246  return status;
247 }

249 /*FUNCTION**********************************************************************
250  *
251  * Function Name : mainFunction
252  * Description : Provides polling support, by handling specific events.
253  *
254  *END**************************************************************************/
255 static status_t mainFunction(uint8_t phy)
256 {
257  status_t status;
258  bool linkUp;
259  phy_basicStatus_t basicStatus;
260 
261  /* Read basic status register */
262  status = PHY_SHARED_ReadBasicStatus(phy, &basicStatus);
263  if (status != STATUS_SUCCESS)
264  {
265  return status;
266  }
267 
268  linkUp = basicStatus.link;
269  linkUp = ((basicStatus.remoteFault) == true) ? (false) : (linkUp);
270 
271  if (linkUp == false)
272  {
273  if ((g_phyConfig[phy].privateData & PRIVATE_LINKUP) != 0U)
274  {
275  /* Link down event */
276  if (g_phyConfig[phy].linkDownEventCB != NULL)
277  {
278  g_phyConfig[phy].linkDownEventCB(phy);
279  }
280  g_phyConfig[phy].privateData = g_phyConfig[phy].privateData & ~PRIVATE_LINKUP;
281  }
282  /* read basic status again, link might be up because of latching function of link status bit */
283  status = PHY_SHARED_ReadBasicStatus(phy, &basicStatus);
284  if (status != STATUS_SUCCESS)
285  {
286  return status;
287  }
288  }
289  if (linkUp == true)
290  {
291  if ((g_phyConfig[phy].privateData & PRIVATE_LINKUP) == 0U)
292  {
293  /* Link changed from down to up */
294  if (g_phyConfig[phy].linkUpEventCB != NULL)
295  {
296  g_phyConfig[phy].linkUpEventCB(phy);
297  }
298  g_phyConfig[phy].privateData = g_phyConfig[phy].privateData | PRIVATE_LINKUP;
299  /* if autoneg enabled, check for status */
300  if ((g_phyConfig[phy].privateData & PRIVATE_ANEG_ENABLED) != 0U)
301  {
302  /* start checking for completion of auto-negotiation */
303  g_phyConfig[phy].privateData = g_phyConfig[phy].privateData | PRIVATE_ANEG_PENDING;
304  }
305  }
306  if ((g_phyConfig[phy].privateData & PRIVATE_ANEG_PENDING) != 0U)
307  {
308  if ((basicStatus.autonegComplete) == true)
309  {
310  /* autoneg complete*/
311  if (g_phyConfig[phy].autonegCompleteCB != NULL)
312  {
313  g_phyConfig[phy].autonegCompleteCB(phy);
314  g_phyConfig[phy].privateData = g_phyConfig[phy].privateData & ~PRIVATE_ANEG_PENDING;
315  }
316  }
317  }
318  }
319 
320  return status;
321 }

487 /*FUNCTION**********************************************************************
488  *
489  * Function Name : getLinkSpeed
490  * Description : Gets the link speed for the specified PHY device.
491  *
492  *END**************************************************************************/
493 static status_t getLinkSpeed(uint8_t phy, phy_speed_t* speed)
494 {
495  status_t status;
496  uint16_t reg;
497  uint16_t autonegCapabilities;
498 
499  if ((g_phyConfig[phy].privateData & PRIVATE_ANEG_ENABLED) == 0U)
500  {
501  /* speed in basic control register only valid when autoneg is disabled */
502  status = PHY_Read(phy, BASIC_CONTROL_ADDR, &reg);
503  if (status != STATUS_SUCCESS)
504  {
505  return status;
506  }
507 
508  if ((reg & BASIC_CONTROL_SPEED_MSB_FLAG) != 0U)
509  {
510  *speed = PHY_SPEED_1_GBPS;
511  if ((reg & BASIC_CONTROL_SPEED_LSB_FLAG) != 0U)
512  {
513  *speed = PHY_SPEED_RESERVED;
514  }
515  }
516  else
517  {
518  *speed = PHY_SPEED_10_MBPS;
519  if ((reg & BASIC_CONTROL_SPEED_LSB_FLAG) != 0U)
520  {
521  *speed = PHY_SPEED_100_MBPS;
522  }
523  }
524  }
525  else
526  {
527  /* auto negotation enabled - check advertised and link partner capabilities */
528  *speed = PHY_SPEED_RESERVED;
529  if ((g_phyConfig[phy].privateData & PRIVATE_1000BASET) != 0U)
530  {
531  status = PHY_Read(phy, CONTROL_1000BASET_ADDR, &autonegCapabilities);
532  if (status != STATUS_SUCCESS)
533  {
534  return status;
535  }
536  status = PHY_Read(phy, STATUS_1000BASET_ADDR, &reg);
537  if (status != STATUS_SUCCESS)
538  {
539  return status;
540  }
541  autonegCapabilities &= (uint16_t) (reg >> CONTROL_STATUS_1000BASET_OFFSET);
542  if ((autonegCapabilities & (AUTONEG_1000BASETFD | AUTONEG_1000BASETHD)) != 0U)
543  {
544  *speed = PHY_SPEED_1_GBPS;
545  }
546  }
547  if (*speed == PHY_SPEED_RESERVED)
548  { /* not gigabit, check if 10 or 100 */
549  status = PHY_Read(phy, AUTONEG_ADVERT_ADDR, &autonegCapabilities);
550  if (status != STATUS_SUCCESS)
551  {
552  return status;
553  }
554  status = PHY_Read(phy, AUTONEG_LPABILITY_ADDR, &reg);
555  if (status != STATUS_SUCCESS)
556  {
557  return status;
558  }
559  autonegCapabilities &= reg;
560  if ((autonegCapabilities & (AUTONEG_100BASETXFD | AUTONEG_100BASETXHD)) != 0U)
561  {
562  *speed = PHY_SPEED_100_MBPS;
563  }
564  else
565  {
566  if ((autonegCapabilities & (AUTONEG_10BASETFD | AUTONEG_10BASETHD)) != 0U)
567  {
568  *speed = PHY_SPEED_10_MBPS;
569  }
570  }
571  }
572  }
573 
574  return STATUS_SUCCESS;
575 }

596 static status_t setRole(uint8_t phy, phy_role_t role)
597 {
598  status_t status;
599  uint16_t reg;
600 
601  if ((g_phyConfig[phy].privateData & PRIVATE_ANEG_MSCTRL) != 0U)
602  {

604  {
605  reg = CONTROL_1000BASET_CONFIG_ENABLE_FLAG;
606  if (role == PHY_ROLE_MASTER)
607  {
608  reg |= CONTROL_1000BASET_CONFIG_VALUE_FLAG;
609  }
610  status = PHY_RMW(phy, CONTROL_1000BASET_ADDR, reg, (uint16_t) CONTROL_1000BASET_CONFIG_ENABLE_FLAG | (uint16_t) CONTROL_1000BASET_CONFIG_VALUE_FLAG);
611  }
612  else
613  {
614  status = PHY_RMW(phy, CONTROL_1000BASET_ADDR, 0, CONTROL_1000BASET_CONFIG_ENABLE_FLAG); /* Deactivate manual configuration */
615  }
616  if (status != STATUS_SUCCESS)
617  {
618  return status;
619  }
620  g_phyConfig[phy].role = role; /* Update role in local shadow */
621  }
622  else
623  {
624  /* No master/slave selection supported */
625  status = STATUS_UNSUPPORTED;
626  }
627 
628  return status;
629 }

659 /*FUNCTION**********************************************************************
660  *
661  * Function Name : checkCapabilities
662  * Description : Checks the capabilities of the specified PHY device.
663  *
664  *END**************************************************************************/
665 static status_t checkCapabilities(uint8_t phy)
666 {
667  status_t status;
668  bool autonegMasterSlave = false;
669  uint16_t reg;
670  phy_basicStatus_t basicStatus;
671 
672  /* Read basic status register */
673  status = PHY_SHARED_ReadBasicStatus(phy, &basicStatus);
674  if (status != STATUS_SUCCESS)
675  {
676  return status;
677  }
678 
679  /* Store Autoneg flag based on capability */
680  g_phyConfig[phy].privateData = ((basicStatus.autonegAbility) == true) ?
681  (g_phyConfig[phy].privateData | PRIVATE_ANEG_CAPABLE) :
682  (g_phyConfig[phy].privateData & ~PRIVATE_ANEG_CAPABLE);
683 
684  if ((basicStatus.baseT2Hd || basicStatus.baseT2Fd) != false)
685  {
686  /* 100BASE-T2 support */
687  autonegMasterSlave = true;
688  }
689 
690  if (basicStatus.extendedStatus == true)
691  {
692  status = PHY_Read(phy, EXTENDED_STATUS_ADDR, &reg);
693  if (status != STATUS_SUCCESS)
694  {
695  return status;
696  }
697  if ((reg & (EXTENDED_STATUS_1000BASETFD_FLAG | EXTENDED_STATUS_1000BASETHD_FLAG)) != 0U)
698  {
699  /* 1000BASE-T support */
700  autonegMasterSlave = true;
701  g_phyConfig[phy].privateData |= PRIVATE_1000BASET;
702  }
703  else
704  {
705  g_phyConfig[phy].privateData &= ~PRIVATE_1000BASET;
706  }
707  }
708 
709  g_phyConfig[phy].privateData = (autonegMasterSlave == false) ?
710  (g_phyConfig[phy].privateData & ~PRIVATE_ANEG_MSCTRL) :
711  (g_phyConfig[phy].privateData | PRIVATE_ANEG_MSCTRL);
712 
713  if ((g_phyConfig[phy].privateData & PRIVATE_ANEG_CAPABLE) != 0U)
714  {
715  /* check default setting for auto-negotiation */
716  status = PHY_Read(phy, BASIC_CONTROL_ADDR, &reg);
717  if (status != STATUS_SUCCESS)
718  {
719  return status;
