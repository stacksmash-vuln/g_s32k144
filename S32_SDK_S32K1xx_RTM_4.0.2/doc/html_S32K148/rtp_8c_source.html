122 static void
123 rtp_send_packets( int sock, struct sockaddr_in* to)
124 {
125  struct rtp_hdr* rtphdr;
126  u8_t* rtp_payload;
127  int rtp_payload_size;
128  size_t rtp_data_index;
129 
130  /* prepare RTP packet */
131  rtphdr = (struct rtp_hdr*)rtp_send_packet;
132  rtphdr->version = RTP_VERSION;
133  rtphdr->payloadtype = 0;
134  rtphdr->ssrc = PP_HTONL(RTP_SSRC);
135  rtphdr->timestamp = lwip_htonl(lwip_ntohl(rtphdr->timestamp) + RTP_TIMESTAMP_INCREMENT);
136 
137  /* send RTP stream packets */
138  rtp_data_index = 0;
139  do {
140  rtp_payload = rtp_send_packet+sizeof(struct rtp_hdr);
141  rtp_payload_size = LWIP_MIN(RTP_PAYLOAD_SIZE, (sizeof(rtp_data) - rtp_data_index));
142 
143  MEMCPY(rtp_payload, rtp_data + rtp_data_index, rtp_payload_size);
144 
145  /* set MARKER bit in RTP header on the last packet of an image */
146  rtphdr->payloadtype = RTP_PAYLOADTYPE | (((rtp_data_index + rtp_payload_size)
147  >= sizeof(rtp_data)) ? RTP_MARKER_MASK : 0);
148 
149  /* send RTP stream packet */
150  if (sendto(sock, rtp_send_packet, sizeof(struct rtp_hdr) + rtp_payload_size,
151  0, (struct sockaddr *)to, sizeof(struct sockaddr)) >= 0) {
152  rtphdr->seqNum = lwip_htons(lwip_ntohs(rtphdr->seqNum) + 1);
153  rtp_data_index += rtp_payload_size;
154  } else {
155  LWIP_DEBUGF(RTP_DEBUG, ("rtp_sender: not sendto==%i\n", errno));
156  }
157  }while (rtp_data_index < sizeof(rtp_data));
158 }

163 static void
164 rtp_send_thread(void *arg)
165 {
166  int sock;
167  struct sockaddr_in local;
168  struct sockaddr_in to;
169  u32_t rtp_stream_address;
170 
171  LWIP_UNUSED_ARG(arg);
172 
173  /* initialize RTP stream address */
174  rtp_stream_address = RTP_STREAM_ADDRESS;
175 
176  /* if we got a valid RTP stream address... */
177  if (rtp_stream_address != 0) {
178  /* create new socket */
179  sock = socket(AF_INET, SOCK_DGRAM, 0);
180  if (sock >= 0) {
181  /* prepare local address */
182  memset(&local, 0, sizeof(local));
183  local.sin_family = AF_INET;
184  local.sin_port = PP_HTONS(INADDR_ANY);
185  local.sin_addr.s_addr = PP_HTONL(INADDR_ANY);
186 
187  /* bind to local address */
188  if (bind(sock, (struct sockaddr *)&local, sizeof(local)) == 0) {
189  /* prepare RTP stream address */
190  memset(&to, 0, sizeof(to));
191  to.sin_family = AF_INET;
192  to.sin_port = PP_HTONS(RTP_STREAM_PORT);
193  to.sin_addr.s_addr = rtp_stream_address;
194 
195  /* send RTP packets */
196  memset(rtp_send_packet, 0, sizeof(rtp_send_packet));
197  while (1) {
198  rtp_send_packets( sock, &to);
199  sys_msleep(RTP_SEND_DELAY);
200  }
201  }
202 
203  /* close the socket */
204  closesocket(sock);
205  }
206  }

212 static void
213 rtp_recv_thread(void *arg)
214 {
215  int sock;
216  struct sockaddr_in local;
217  struct sockaddr_in from;
218  int fromlen;
219  struct ip_mreq ipmreq;
220  struct rtp_hdr* rtphdr;
221  u32_t rtp_stream_address;
222  int timeout;
223  size_t result;
224  int recvrtppackets = 0;
225  int lostrtppackets = 0;
226  u16_t lastrtpseq = 0;
227 
228  LWIP_UNUSED_ARG(arg);
229 
230  /* initialize RTP stream address */
231  rtp_stream_address = RTP_STREAM_ADDRESS;
232 
233  /* if we got a valid RTP stream address... */
234  if (rtp_stream_address != 0) {
235  /* create new socket */
236  sock = socket(AF_INET, SOCK_DGRAM, 0);
237  if (sock >= 0) {
238  /* prepare local address */
239  memset(&local, 0, sizeof(local));
240  local.sin_family = AF_INET;
241  local.sin_port = PP_HTONS(RTP_STREAM_PORT);
242  local.sin_addr.s_addr = PP_HTONL(INADDR_ANY);
243 
244  /* bind to local address */
245  if (bind(sock, (struct sockaddr *)&local, sizeof(local)) == 0) {
246  /* set recv timeout */
247  timeout = RTP_RECV_TIMEOUT;
248  setsockopt(sock, SOL_SOCKET, SO_RCVTIMEO, (char *)&timeout, sizeof(timeout));
249 
250  /* prepare multicast "ip_mreq" struct */
251  ipmreq.imr_multiaddr.s_addr = rtp_stream_address;
252  ipmreq.imr_interface.s_addr = PP_HTONL(INADDR_ANY);
253 
254  /* join multicast group */
255  if (setsockopt(sock, IPPROTO_IP, IP_ADD_MEMBERSHIP, &ipmreq, sizeof(ipmreq)) == 0) {
256  /* receive RTP packets */
257  while(1) {
258  fromlen = sizeof(from);
259  result = recvfrom(sock, rtp_recv_packet, sizeof(rtp_recv_packet), 0,
260  (struct sockaddr *)&from, (socklen_t *)&fromlen);
261  if (result >= sizeof(struct rtp_hdr)) {
262  rtphdr = (struct rtp_hdr *)rtp_recv_packet;
263  recvrtppackets++;
264  if ((lastrtpseq == 0) || ((lastrtpseq + 1) == lwip_ntohs(rtphdr->seqNum))) {
265  RTP_RECV_PROCESSING((rtp_recv_packet + sizeof(rtp_hdr)),(result-sizeof(rtp_hdr)));
266  } else {
267  lostrtppackets++;
268  }
269  lastrtpseq = lwip_ntohs(rtphdr->seqNum);
270  if ((recvrtppackets % RTP_RECV_STATS) == 0) {
271  LWIP_DEBUGF(RTP_DEBUG, ("rtp_recv_thread: recv %6i packet(s) / lost %4i packet(s) (%.4f%%)...\n", recvrtppackets, lostrtppackets, (lostrtppackets*100.0)/recvrtppackets));
272  }
273  } else {
274  LWIP_DEBUGF(RTP_DEBUG, ("rtp_recv_thread: recv timeout...\n"));
275  }
276  }
277 
278  /* leave multicast group */
279  setsockopt(sock, IPPROTO_IP, IP_DROP_MEMBERSHIP, &ipmreq, sizeof(ipmreq));
280  }
281  }
282 
283  /* close the socket */
284  closesocket(sock);
285  }
286  }
287 }
