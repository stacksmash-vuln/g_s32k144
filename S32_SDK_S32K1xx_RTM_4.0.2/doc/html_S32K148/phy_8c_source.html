410 /*FUNCTION**********************************************************************
411  *
412  * Function Name : PHY_RMW
413  * Description : Reads and modifies a register of the specified PHY device.
414  * Implements : PHY_RMW_Activity
415  *
416  *END**************************************************************************/
417 status_t PHY_RMW(uint8_t phy, uint8_t registerAddr, uint16_t value, uint16_t mask)
418 {
419  status_t status;
420  uint16_t reg;
421 
422  if (s_phyDrivers[g_phyConfig[phy].type]->rmw != NULL)
423  { /* driver has a custom read modify write function */
424  status = s_phyDrivers[g_phyConfig[phy].type]->rmw(phy, registerAddr, value, mask);
425  }
426  else
427  {
428  /* Read */
429  status = PHY_Read(phy, registerAddr, &reg);
430  if (status == STATUS_SUCCESS)
431  {
432  /* Modify */
433  reg &= (uint16_t)~mask; /* Clear masked bits */
434  reg |= (uint16_t)(mask & value); /* Set masked bits to value */
435 
436  /* Write */
437  status = PHY_Write(phy, registerAddr, reg);
438  }
439  }
440 
441  return status;
442 }

444 /*FUNCTION**********************************************************************
445  *
446  * Function Name : PHY_WriteMMD
447  * Description : Writes a register of the specified MMD in a PHY device.
448  * Implements : PHY_WriteMMD_Activity
449  *
450  *END**************************************************************************/
451 status_t PHY_WriteMMD(uint8_t phy, uint8_t mmd, uint16_t registerAddr, uint16_t value)
452 {
453  status_t status;
454 
455  if ((s_phyDrivers[g_phyConfig[phy].type]->writeMMD != NULL) && (g_phyConfig[phy].location == PHY_LOCATION_INTERNAL))
456  { /* driver has a custom MMD write function for internal access */
457  status = s_phyDrivers[g_phyConfig[phy].type]->writeMMD(phy, mmd, registerAddr, value);
458  }
459  else
460  {
461  if ((g_phyConfig[phy].location == PHY_LOCATION_EXTERNAL) && (g_phyConfig[phy].clause45 == true))
462  {
463  status = PHY_HW_WriteCl45(g_phyConfig[phy].enetInstance, (uint8_t) g_phyConfig[phy].addr, mmd, registerAddr, value, PHY_MDIO_TIMEOUT);
464  }
465  else
466  {
467  status = writeCl22(phy, MMD_ACCESS_CONTROL_ADDR, mmd);
468  if (status != STATUS_SUCCESS)
469  {
470  return status;
471  }
472  status = writeCl22(phy, MMD_ACCESS_REG_DATA_ADDR, registerAddr);
473  if (status != STATUS_SUCCESS)
474  {
475  return status;
476  }
477  status = writeCl22(phy, MMD_ACCESS_CONTROL_ADDR, MMD_ACCESS_CONTROL_DATA | mmd);
478  if (status != STATUS_SUCCESS)
479  {
480  return status;
481  }
482  status = writeCl22(phy, MMD_ACCESS_REG_DATA_ADDR, value);
483  }
484 
485  }
486 
487  return status;
488 }

570  * Function Name : writeCl22
571  * Description : Writes a register of the specified PHY device.
572  *
573  *END**************************************************************************/
574 static status_t writeCl22(uint8_t phy, uint8_t registerAddr, uint16_t value)
575 {
576  status_t status = STATUS_SUCCESS;
577 
578  if (g_phyConfig[phy].location == PHY_LOCATION_INTERNAL)
579  {
580  ((volatile uint16_t *)g_phyConfig[phy].addr)[(uint32_t) ((uint32_t) registerAddr << ALIGN32_SHIFT)] = value;
581  }
582  else
583  {
584  if (PHY_HW_WriteCl22(g_phyConfig[phy].enetInstance, (uint8_t)g_phyConfig[phy].addr, registerAddr, value, PHY_MDIO_TIMEOUT) != STATUS_SUCCESS)
585  {
586  status = STATUS_PHY_ACCESS_FAILED;
587  }
588  }
589 
590  return status;
591 }

593 /*FUNCTION**********************************************************************

598  *END**************************************************************************/
599 static status_t readCl22(uint8_t phy, uint8_t registerAddr, uint16_t* value)
600 {
601  status_t status = STATUS_SUCCESS;
602  if (g_phyConfig[phy].location == PHY_LOCATION_INTERNAL)
603  {
604  *value = ((volatile uint16_t *)g_phyConfig[phy].addr)[(uint32_t) ((uint32_t) registerAddr << ALIGN32_SHIFT)];
605  }
606  else
607  {
608  if (PHY_HW_ReadCl22(g_phyConfig[phy].enetInstance, (uint8_t)g_phyConfig[phy].addr, registerAddr, value, PHY_MDIO_TIMEOUT) != STATUS_SUCCESS)
609  {
610  status = STATUS_PHY_ACCESS_FAILED;
611  }
612  }
613  return status;
614 }
