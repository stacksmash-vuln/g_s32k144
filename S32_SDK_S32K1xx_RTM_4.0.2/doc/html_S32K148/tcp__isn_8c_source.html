116 u32_t
117 lwip_hook_tcp_isn(const ip_addr_t *local_ip, u16_t local_port,
118  const ip_addr_t *remote_ip, u16_t remote_port)
119 {
120  md5_context ctx;
121  u8_t output[16];
122  u32_t isn;
123 
124 #if LWIP_IPV4 && LWIP_IPV6
125  if (IP_IS_V6(local_ip))
126 #endif /* LWIP_IPV4 && LWIP_IPV6 */
127 #if LWIP_IPV6
128  {
129  const ip6_addr_t *local_ip6, *remote_ip6;
130 
131  local_ip6 = ip_2_ip6(local_ip);
132  remote_ip6 = ip_2_ip6(remote_ip);
133 
134  SMEMCPY(&input[0], &local_ip6->addr, 16);
135  SMEMCPY(&input[16], &remote_ip6->addr, 16);
136  }
137 #endif /* LWIP_IPV6 */
138 #if LWIP_IPV4 && LWIP_IPV6
139  else
140 #endif /* LWIP_IPV4 && LWIP_IPV6 */
141 #if LWIP_IPV4
142  {
143  const ip4_addr_t *local_ip4, *remote_ip4;
144 
145  local_ip4 = ip_2_ip4(local_ip);
146  remote_ip4 = ip_2_ip4(remote_ip);
147 
148  /* Represent IPv4 addresses as IPv4-mapped IPv6 addresses, to ensure that
149  * the IPv4 and IPv6 address spaces are completely disjoint. */
150  memset(&input[0], 0, 10);
151  input[10] = 0xff;
152  input[11] = 0xff;
153  SMEMCPY(&input[12], &local_ip4->addr, 4);
154  memset(&input[16], 0, 10);
155  input[26] = 0xff;
156  input[27] = 0xff;
157  SMEMCPY(&input[28], &remote_ip4->addr, 4);
158  }
159 #endif /* LWIP_IPV4 */
160 
161  input[32] = local_port >> 8;
162  input[33] = local_port & 0xff;
163  input[34] = remote_port >> 8;
164  input[35] = remote_port & 0xff;
165 
166  /* The secret and padding are already filled in. */
167 
168  /* Generate the hash, using MD5. */
169  md5_starts(&ctx);
170  md5_update(&ctx, input, sizeof(input));
171  md5_finish(&ctx, output);
172 
173  /* Arbitrarily take the first 32 bits from the generated hash. */
174  MEMCPY(&isn, output, sizeof(isn));
175 
176  /* Add the current time in 4-microsecond units. */
177  return isn + base_time + sys_now() * 250;
178 }
