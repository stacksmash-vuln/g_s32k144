455 /*FUNCTION**********************************************************************
456  *
457  * Function Name : SAI_DRV_TxMuxLineInterrupt
458  * Description : Manage interrupt transfer, separated from original txIrq
459  * to reduce cyclomatic complexity
460  *
461  *END**************************************************************************/
462 static void SAI_DRV_TxMuxLineInterrupt(uint32_t instNum,
463  SAI_Type* inst,
464  sai_state_t* pState,
465  bool* finish)
466 {
467  uint8_t i,j;
468  uint32_t temp;
469 
470  for (i = 0U; i < sai_channel_count[instNum]; i++)
471  {
472  if (SAI_DRV_IsTxChannelEnabled(inst, i))
473  {
474  if (pState->ChnState[0].count < TX_FIFO_SIZE)
475  {
476  temp = pState->ChnState[0].count;
477  }
478  else
479  {
480  temp = TX_FIFO_SIZE;
481  }
482  for (j = 0U; j < temp; j++)
483  {
484  switch (pState->ElementSize)
485  {
486  case 1U:
487  SAI_DRV_TxWrite(inst, i, (uint32_t)*pState->ChnState[pState->NextChn].data);
488  break;
489  case 2U:
490  SAI_DRV_TxWrite(inst, i, (uint32_t)*((uint16_t*) pState->ChnState[pState->NextChn].data));
491  break;
492  case 4U:
493  SAI_DRV_TxWrite(inst, i, *((uint32_t*) pState->ChnState[pState->NextChn].data));
494  break;
495  default:
496  DEV_ASSERT(false);
497  break;
498  }
499  pState->ChnState[pState->NextChn].data += pState->ElementSize;
500  pState->NextChn++; /* alternate between data block */
501  if (pState->NextChn == pState->ChannelCount)
502  {
503  pState->NextChn = 0U;
504  }
505  }
506  if (pState->ChnState[0].count <= TX_FIFO_SIZE)
507  {
508  *finish = true;
509  pState->ChnState[0].count = 0;
510  }
511  else
512  {
513  pState->ChnState[0].count -= TX_FIFO_SIZE;
514  }
515  break; /* only one channle enabled in this mode */
516  }
517  }
518 }
