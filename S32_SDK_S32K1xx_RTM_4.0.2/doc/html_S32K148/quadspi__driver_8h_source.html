332 static inline void QSPI_DRV_SetLut(uint32_t instance,
333  uint8_t lut,
334  qspi_lut_commands_t instr0,
335  qspi_lut_pads_t pad0,
336  uint8_t oprnd0,
337  qspi_lut_commands_t instr1,
338  qspi_lut_pads_t pad1,
339  uint8_t oprnd1)
340 {
341  QuadSPI_Type *baseAddr;
342  DEV_ASSERT(instance < QuadSPI_INSTANCE_COUNT);
343  DEV_ASSERT(lut < QuadSPI_LUT_COUNT);
344 
345  baseAddr = g_qspiBase[instance];
346  baseAddr->LUT[lut] = QuadSPI_LUT_INSTR0(instr0)
347  | QuadSPI_LUT_PAD0(pad0)
348  | QuadSPI_LUT_OPRND0(oprnd0)
349  | QuadSPI_LUT_INSTR1(instr1)
350  | QuadSPI_LUT_PAD1(pad1)
351  | QuadSPI_LUT_OPRND1(oprnd1);
352 }
