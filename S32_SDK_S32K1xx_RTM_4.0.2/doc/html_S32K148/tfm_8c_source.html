78 void fp_add(fp_int *a, fp_int *b, fp_int *c)
79 {
80  int sa, sb;
81 
82  /* get sign of both inputs */
83  sa = a->sign;
84  sb = b->sign;
85 
86  /* handle two cases, not four */
87  if (sa == sb) {
88  /* both positive or both negative */
89  /* add their magnitudes, copy the sign */
90  c->sign = sa;
91  s_fp_add (a, b, c);
92  } else {
93  /* one positive, the other negative */
94  /* subtract the one with the greater magnitude from */
95  /* the one of the lesser magnitude. The result gets */
96  /* the sign of the one with the greater magnitude. */
97  if (fp_cmp_mag (a, b) == FP_LT) {
98  c->sign = sb;
99  s_fp_sub (b, a, c);
100  } else {
101  c->sign = sa;
102  s_fp_sub (a, b, c);
103  }
104  }

108 void s_fp_add(fp_int *a, fp_int *b, fp_int *c)
109 {
110  int x, y, oldused;
111  fp_word t;
112 
113  y = MAX(a->used, b->used);
114  oldused = MIN(c->used, FP_SIZE); /* help static analysis w/ largest size */
115  c->used = y;
116 
117  t = 0;

119  t += ((fp_word)a->dp[x]) + ((fp_word)b->dp[x]);

137 /* c = a - b */
138 void fp_sub(fp_int *a, fp_int *b, fp_int *c)
139 {
140  int sa, sb;
141 
142  sa = a->sign;
143  sb = b->sign;
144 
145  if (sa != sb) {
146  /* subtract a negative from a positive, OR */
147  /* subtract a positive from a negative. */
148  /* In either case, ADD their magnitudes, */
149  /* and use the sign of the first number. */
150  c->sign = sa;
151  s_fp_add (a, b, c);
152  } else {
153  /* subtract a positive from a positive, OR */
154  /* subtract a negative from a negative. */
155  /* First, take the difference between their */
156  /* magnitudes, then... */
157  if (fp_cmp_mag (a, b) != FP_LT) {
158  /* Copy the sign from the first */
159  c->sign = sa;
160  /* The first has a larger or equal magnitude */
161  s_fp_sub (a, b, c);
162  } else {
163  /* The result has the *opposite* sign from */
164  /* the first number. */
165  c->sign = (sa == FP_ZPOS) ? FP_NEG : FP_ZPOS;
166  /* The second has a larger magnitude */
167  s_fp_sub (b, a, c);
168  }
169  }

322 void fp_mul_2(fp_int * a, fp_int * b)
323 {
324  int x, oldused;
325 
326  oldused = b->used;
327  b->used = a->used;
328 
329  {
330  fp_digit r, rr, *tmpa, *tmpb;
331 
332  /* alias for source */
333  tmpa = a->dp;
334 
335  /* alias for dest */
336  tmpb = b->dp;
337 
338  /* carry */
339  r = 0;
340  for (x = 0; x < a->used; x++) {
341 
342  /* get what will be the *next* carry bit from the
343  * MSB of the current digit
344  */
345  rr = *tmpa >> ((fp_digit)(DIGIT_BIT - 1));
346 
347  /* now shift up this digit, add in the carry [from the previous] */
348  *tmpb++ = ((*tmpa++ << ((fp_digit)1)) | r);
349 
350  /* copy the carry that would be from the source
351  * digit into the next iteration
352  */
353  r = rr;
354  }
355 
356  /* new leading digit? */
357  if (r != 0 && b->used != (FP_SIZE-1)) {
358  /* add a MSB which is always 1 at this point */
359  *tmpb = 1;
360  ++(b->used);
361  }
362 
363  /* zero any excess digits on the destination that we didn't write to */
364  tmpb = b->dp + b->used;
365  for (x = b->used; x < oldused; x++) {
366  *tmpb++ = 0;
367  }
368  }
369  b->sign = a->sign;
370 }

399 /* c = a * 2**d */
400 void fp_mul_2d(fp_int *a, int b, fp_int *c)
401 {
402  fp_digit carry, carrytmp, shift;
403  int x;
404 
405  /* copy it */
406  fp_copy(a, c);
407 
408  /* handle whole digits */
409  if (b >= DIGIT_BIT) {
410  fp_lshd(c, b/DIGIT_BIT);
411  }
412  b %= DIGIT_BIT;
413 
414  /* shift the digits */
415  if (b != 0) {
416  carry = 0;
417  shift = DIGIT_BIT - b;
418  for (x = 0; x < c->used; x++) {
419  carrytmp = c->dp[x] >> shift;
420  c->dp[x] = (c->dp[x] << b) + carry;
421  carry = carrytmp;
422  }
423  /* store last carry if room */
424  if (carry && x < FP_SIZE) {
425  c->dp[c->used++] = carry;
426  }
427  }
428  fp_clamp(c);
429 }

519 /* a/b => cb + d == a */
520 int fp_div(fp_int *a, fp_int *b, fp_int *c, fp_int *d)
521 {
522  fp_int q, x, y, t1, t2;
523  int n, t, i, norm, neg;
524 
525  /* is divisor zero ? */
526  if (fp_iszero (b) == FP_YES) {
527  return FP_VAL;
528  }
529 
530  /* if a < b then q=0, r = a */
531  if (fp_cmp_mag (a, b) == FP_LT) {
532  if (d != NULL) {
533  fp_copy (a, d);
534  }
535  if (c != NULL) {
536  fp_zero (c);
537  }
538  return FP_OKAY;
539  }
540 
541  fp_init(&q);
542  q.used = a->used + 2;
543 
544  fp_init(&t1);
545  fp_init(&t2);
546  fp_init_copy(&x, a);
547  fp_init_copy(&y, b);
548 
549  /* fix the sign */
550  neg = (a->sign == b->sign) ? FP_ZPOS : FP_NEG;
551  x.sign = y.sign = FP_ZPOS;
552 
553  /* normalize both x and y, ensure that y >= b/2, [b == 2**DIGIT_BIT] */
554  norm = fp_count_bits(&y) % DIGIT_BIT;
555  if (norm < (int)(DIGIT_BIT-1)) {
556  norm = (DIGIT_BIT-1) - norm;
557  fp_mul_2d (&x, norm, &x);
558  fp_mul_2d (&y, norm, &y);
559  } else {
560  norm = 0;
561  }
562 
563  /* note hac does 0 based, so if used==5 then its 0,1,2,3,4, e.g. use 4 */
564  n = x.used - 1;
565  t = y.used - 1;
566 
567  /* while (x >= y*b**n-t) do { q[n-t] += 1; x -= y*b**{n-t} } */
568  fp_lshd (&y, n - t); /* y = y*b**{n-t} */
569 
570  while (fp_cmp (&x, &y) != FP_LT) {
571  ++(q.dp[n - t]);
572  fp_sub (&x, &y, &x);
573  }
574 
575  /* reset y by shifting it back down */
576  fp_rshd (&y, n - t);
577 
578  /* step 3. for i from n down to (t + 1) */
579  for (i = n; i >= (t + 1); i--) {
580  if (i > x.used) {
581  continue;
582  }
583 
584  /* step 3.1 if xi == yt then set q{i-t-1} to b-1,
585  * otherwise set q{i-t-1} to (xi*b + x{i-1})/yt */
586  if (x.dp[i] == y.dp[t]) {
587  q.dp[i - t - 1] = (fp_digit) ((((fp_word)1) << DIGIT_BIT) - 1);
588  } else {
589  fp_word tmp;
590  tmp = ((fp_word) x.dp[i]) << ((fp_word) DIGIT_BIT);
591  tmp |= ((fp_word) x.dp[i - 1]);
592  tmp /= ((fp_word)y.dp[t]);
593  q.dp[i - t - 1] = (fp_digit) (tmp);
594  }
595 
596  /* while (q{i-t-1} * (yt * b + y{t-1})) >
597  xi * b**2 + xi-1 * b + xi-2
598 
599  do q{i-t-1} -= 1;
600  */
601  q.dp[i - t - 1] = (q.dp[i - t - 1] + 1);
602  do {
603  q.dp[i - t - 1] = (q.dp[i - t - 1] - 1);
604 
605  /* find left hand */
606  fp_zero (&t1);
607  t1.dp[0] = (t - 1 < 0) ? 0 : y.dp[t - 1];
608  t1.dp[1] = y.dp[t];
609  t1.used = 2;
610  fp_mul_d (&t1, q.dp[i - t - 1], &t1);
611 
612  /* find right hand */
613  t2.dp[0] = (i - 2 < 0) ? 0 : x.dp[i - 2];
614  t2.dp[1] = (i - 1 < 0) ? 0 : x.dp[i - 1];
615  t2.dp[2] = x.dp[i];
616  t2.used = 3;
617  } while (fp_cmp_mag(&t1, &t2) == FP_GT);
618 
619  /* step 3.3 x = x - q{i-t-1} * y * b**{i-t-1} */
620  fp_mul_d (&y, q.dp[i - t - 1], &t1);
621  fp_lshd (&t1, i - t - 1);
622  fp_sub (&x, &t1, &x);
623 
624  /* if x < 0 then { x = x + y*b**{i-t-1}; q{i-t-1} -= 1; } */
625  if (x.sign == FP_NEG) {
626  fp_copy (&y, &t1);
627  fp_lshd (&t1, i - t - 1);
628  fp_add (&x, &t1, &x);
629  q.dp[i - t - 1] = q.dp[i - t - 1] - 1;
630  }
631  }
632 
633  /* now q is the quotient and x is the remainder
634  * [which we have to normalize]
635  */
636 
637  /* get sign before writing to c */
638  x.sign = x.used == 0 ? FP_ZPOS : a->sign;
639 
640  if (c != NULL) {
641  fp_clamp (&q);
642  fp_copy (&q, c);
643  c->sign = neg;
644  }
645 
646  if (d != NULL) {
647  fp_div_2d (&x, norm, &x, NULL);
648 
649  /* zero any excess digits on the destination that we didn't write to */
650  for (i = b->used; i < x.used; i++) {
651  x.dp[i] = 0;
652  }
653  fp_clamp(&x);
654  fp_copy (&x, d);
655  }
656 
657  return FP_OKAY;
658 }

786 static int fp_invmod_slow (fp_int * a, fp_int * b, fp_int * c)
787 {
788  fp_int x, y, u, v, A, B, C, D;
789  int res;
790 
791  /* b cannot be negative */
792  if (b->sign == FP_NEG || fp_iszero(b) == FP_YES) {
793  return FP_VAL;
794  }
795 
796  /* init temps */
797  fp_init(&x); fp_init(&y);
798  fp_init(&u); fp_init(&v);
799  fp_init(&A); fp_init(&B);
800  fp_init(&C); fp_init(&D);
801 
802  /* x = a, y = b */
803  if ((res = fp_mod(a, b, &x)) != FP_OKAY) {
804  return res;
805  }
806  fp_copy(b, &y);
807 
808  /* 2. [modified] if x,y are both even then return an error! */
809  if (fp_iseven (&x) == FP_YES && fp_iseven (&y) == FP_YES) {
810  return FP_VAL;
811  }
812 
813  /* 3. u=x, v=y, A=1, B=0, C=0,D=1 */
814  fp_copy (&x, &u);
815  fp_copy (&y, &v);
816  fp_set (&A, 1);
817  fp_set (&D, 1);
818 
819 top:
820  /* 4. while u is even do */
821  while (fp_iseven (&u) == FP_YES) {
822  /* 4.1 u = u/2 */
823  fp_div_2 (&u, &u);
824 
825  /* 4.2 if A or B is odd then */
826  if (fp_isodd (&A) == FP_YES || fp_isodd (&B) == FP_YES) {
827  /* A = (A+y)/2, B = (B-x)/2 */
828  fp_add (&A, &y, &A);
829  fp_sub (&B, &x, &B);
830  }
831  /* A = A/2, B = B/2 */
832  fp_div_2 (&A, &A);
833  fp_div_2 (&B, &B);
834  }
835 
836  /* 5. while v is even do */
837  while (fp_iseven (&v) == FP_YES) {
838  /* 5.1 v = v/2 */
839  fp_div_2 (&v, &v);
840 
841  /* 5.2 if C or D is odd then */
842  if (fp_isodd (&C) == FP_YES || fp_isodd (&D) == FP_YES) {
843  /* C = (C+y)/2, D = (D-x)/2 */
844  fp_add (&C, &y, &C);
845  fp_sub (&D, &x, &D);
846  }
847  /* C = C/2, D = D/2 */
848  fp_div_2 (&C, &C);
849  fp_div_2 (&D, &D);
850  }
851 
852  /* 6. if u >= v then */
853  if (fp_cmp (&u, &v) != FP_LT) {
854  /* u = u - v, A = A - C, B = B - D */
855  fp_sub (&u, &v, &u);
856  fp_sub (&A, &C, &A);
857  fp_sub (&B, &D, &B);
858  } else {
859  /* v - v - u, C = C - A, D = D - B */
860  fp_sub (&v, &u, &v);
861  fp_sub (&C, &A, &C);
862  fp_sub (&D, &B, &D);
863  }
864 
865  /* if not zero goto step 4 */
866  if (fp_iszero (&u) == FP_NO)
867  goto top;
868 
869  /* now a = C, b = D, gcd == g*v */
870 
871  /* if v != 1 then there is no inverse */
872  if (fp_cmp_d (&v, 1) != FP_EQ) {
873  return FP_VAL;
874  }
875 
876  /* if its too low */
877  while (fp_cmp_d(&C, 0) == FP_LT) {
878  fp_add(&C, b, &C);
879  }
880 
881  /* too big */
882  while (fp_cmp_mag(&C, b) != FP_LT) {
883  fp_sub(&C, b, &C);
884  }
885 
886  /* C is now the inverse */
887  fp_copy(&C, c);
888  return FP_OKAY;
889 }

1136 /* y = g**x (mod b)
1137  * Some restrictions... x must be positive and < b
1138  */
1139 static int _fp_exptmod(fp_int * G, fp_int * X, fp_int * P, fp_int * Y)
1140 {
1141  fp_int res;
1142  fp_digit buf, mp;
1143  int err, bitbuf, bitcpy, bitcnt, mode, digidx, x, y, winsize;
1144 #ifdef WOLFSSL_SMALL_STACK
1145  fp_int *M;
1146 #else
1147  fp_int M[64];
1148 #endif
1149 
1150  /* find window size */
1151  x = fp_count_bits (X);
1152  if (x <= 21) {
1153  winsize = 1;
1154  } else if (x <= 36) {
1155  winsize = 3;
1156  } else if (x <= 140) {
1157  winsize = 4;
1158  } else if (x <= 450) {
1159  winsize = 5;
1160  } else {
1161  winsize = 6;
1162  }
1163 
1164  /* now setup montgomery */
1165  if ((err = fp_montgomery_setup (P, &mp)) != FP_OKAY) {
1166  return err;
1167  }
1168 
1169 #ifdef WOLFSSL_SMALL_STACK
1170  /* only allocate space for what's needed */
1171  M = (fp_int*)XMALLOC(sizeof(fp_int)*(1 << winsize), NULL, DYNAMIC_TYPE_TMP_BUFFER);
1172  if (M == NULL) {
1173  return FP_MEM;
1174  }
1175 #endif
1176 
1177  /* init M array */
1178  for(x = 0; x < (1 << winsize); x++)
1179  fp_init(&M[x]);
1180 
1181  /* setup result */
1182  fp_init(&res);
1183 
1184  /* create M table
1185  *
1186  * The M table contains powers of the input base, e.g. M[x] = G^x mod P
1187  *
1188  * The first half of the table is not computed though except for M[0] and M[1]
1189  */
1190 
1191  /* now we need R mod m */
1192  fp_montgomery_calc_normalization (&res, P);
1193 
1194  /* now set M[1] to G * R mod m */
1195  if (fp_cmp_mag(P, G) != FP_GT) {
1196  /* G > P so we reduce it first */
1197  fp_mod(G, P, &M[1]);
1198  } else {
1199  fp_copy(G, &M[1]);
1200  }
1201  fp_mulmod (&M[1], &res, P, &M[1]);
1202 
1203  /* compute the value at M[1<<(winsize-1)] by
1204  * squaring M[1] (winsize-1) times */
1205  fp_copy (&M[1], &M[1 << (winsize - 1)]);
1206  for (x = 0; x < (winsize - 1); x++) {
1207  fp_sqr (&M[1 << (winsize - 1)], &M[1 << (winsize - 1)]);
1208  fp_montgomery_reduce (&M[1 << (winsize - 1)], P, mp);
1209  }
1210 
1211  /* create upper table */
1212  for (x = (1 << (winsize - 1)) + 1; x < (1 << winsize); x++) {
1213  fp_mul(&M[x - 1], &M[1], &M[x]);
1214  fp_montgomery_reduce(&M[x], P, mp);
1215  }
1216 
1217  /* set initial mode and bit cnt */
1218  mode = 0;
1219  bitcnt = 1;
1220  buf = 0;
1221  digidx = X->used - 1;
1222  bitcpy = 0;
1223  bitbuf = 0;
1224 
1225  for (;;) {
1226  /* grab next digit as required */
1227  if (--bitcnt == 0) {
1228  /* if digidx == -1 we are out of digits so break */
1229  if (digidx == -1) {
1230  break;
1231  }
1232  /* read next digit and reset bitcnt */
1233  buf = X->dp[digidx--];
1234  bitcnt = (int)DIGIT_BIT;
1235  }
1236 
1237  /* grab the next msb from the exponent */
1238  y = (int)(buf >> (DIGIT_BIT - 1)) & 1;
1239  buf <<= (fp_digit)1;
1240 
1241  /* if the bit is zero and mode == 0 then we ignore it
1242  * These represent the leading zero bits before the first 1 bit
1243  * in the exponent. Technically this opt is not required but it
1244  * does lower the # of trivial squaring/reductions used
1245  */
1246  if (mode == 0 && y == 0) {
1247  continue;
1248  }
1249 
1250  /* if the bit is zero and mode == 1 then we square */
1251  if (mode == 1 && y == 0) {
1252  fp_sqr(&res, &res);
1253  fp_montgomery_reduce(&res, P, mp);
1254  continue;
1255  }
1256 
1257  /* else we add it to the window */
1258  bitbuf |= (y << (winsize - ++bitcpy));
1259  mode = 2;
1260 
1261  if (bitcpy == winsize) {
1262  /* ok window is filled so square as required and multiply */
1263  /* square first */
1264  for (x = 0; x < winsize; x++) {
1265  fp_sqr(&res, &res);
1266  fp_montgomery_reduce(&res, P, mp);
1267  }
1268 
1269  /* then multiply */
1270  fp_mul(&res, &M[bitbuf], &res);
1271  fp_montgomery_reduce(&res, P, mp);
1272 
1273  /* empty window and reset */
1274  bitcpy = 0;
1275  bitbuf = 0;
1276  mode = 1;
1277  }
1278  }
1279 
1280  /* if bits remain then square/multiply */
1281  if (mode == 2 && bitcpy > 0) {
1282  /* square then multiply if the bit is set */
1283  for (x = 0; x < bitcpy; x++) {
1284  fp_sqr(&res, &res);
1285  fp_montgomery_reduce(&res, P, mp);
1286 
1287  /* get next bit of the window */
1288  bitbuf <<= 1;
1289  if ((bitbuf & (1 << winsize)) != 0) {
1290  /* then multiply */
1291  fp_mul(&res, &M[1], &res);
1292  fp_montgomery_reduce(&res, P, mp);
1293  }
1294  }
1295  }
1296 
1297  /* fixup result if Montgomery reduction is used
1298  * recall that any value in a Montgomery system is
1299  * actually multiplied by R mod n. So we have
1300  * to reduce one more time to cancel out the factor
1301  * of R.
1302  */
1303  fp_montgomery_reduce(&res, P, mp);
1304 
1305  /* swap res with Y */
1306  fp_copy (&res, Y);
1307 
1308 #ifdef WOLFSSL_SMALL_STACK
1309  XFREE(M, NULL, DYNAMIC_TYPE_TMP_BUFFER);
1310 #endif
1311 
1312  return FP_OKAY;
1313 }
