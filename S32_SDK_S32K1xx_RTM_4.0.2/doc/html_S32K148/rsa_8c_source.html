387 /* Uses MGF1 standard as a mask generation function
388  hType: hash type used
389  seed: seed to use for generating mask
390  seedSz: size of seed buffer
391  out: mask output after generation
392  outSz: size of output buffer
393  */
394 static int RsaMGF1(enum wc_HashType hType, byte* seed, word32 seedSz,
395  byte* out, word32 outSz, void* heap)
396 {
397  byte* tmp;
398  /* needs to be large enough for seed size plus counter(4) */
399  byte tmpA[WC_MAX_DIGEST_SIZE + 4];
400  byte tmpF; /* 1 if dynamic memory needs freed */
401  word32 tmpSz;
402  int hLen;
403  int ret;
404  word32 counter;
405  word32 idx;
406  hLen = wc_HashGetDigestSize(hType);
407  counter = 0;
408  idx = 0;
409 
410  (void)heap;
411 
412  /* check error return of wc_HashGetDigestSize */
413  if (hLen < 0) {
414  return hLen;
415  }
416 
417  /* if tmp is not large enough than use some dynamic memory */
418  if ((seedSz + 4) > sizeof(tmpA) || (word32)hLen > sizeof(tmpA)) {
419  /* find largest amount of memory needed which will be the max of
420  * hLen and (seedSz + 4) since tmp is used to store the hash digest */
421  tmpSz = ((seedSz + 4) > (word32)hLen)? seedSz + 4: (word32)hLen;
422  tmp = (byte*)XMALLOC(tmpSz, heap, DYNAMIC_TYPE_RSA_BUFFER);
423  if (tmp == NULL) {
424  return MEMORY_E;
425  }
426  tmpF = 1; /* make sure to free memory when done */
427  }
428  else {
429  /* use array on the stack */
430  tmpSz = sizeof(tmpA);
431  tmp = tmpA;
432  tmpF = 0; /* no need to free memory at end */
433  }
434 
435  do {
436  int i = 0;
437  XMEMCPY(tmp, seed, seedSz);
438 
439  /* counter to byte array appended to tmp */
440  tmp[seedSz] = (counter >> 24) & 0xFF;
441  tmp[seedSz + 1] = (counter >> 16) & 0xFF;
442  tmp[seedSz + 2] = (counter >> 8) & 0xFF;
443  tmp[seedSz + 3] = (counter) & 0xFF;
444 
445  /* hash and append to existing output */
446  if ((ret = wc_Hash(hType, tmp, (seedSz + 4), tmp, tmpSz)) != 0) {
447  /* check for if dynamic memory was needed, then free */
448  if (tmpF) {
449  XFREE(tmp, heap, DYNAMIC_TYPE_RSA_BUFFER);
450  }
451  return ret;
452  }
453 
454  for (i = 0; i < hLen && idx < outSz; i++) {
455  out[idx++] = tmp[i];
456  }
457  counter++;
458  } while (idx < outSz);
459 
460  /* check for if dynamic memory was needed, then free */
461  if (tmpF) {
462  XFREE(tmp, heap, DYNAMIC_TYPE_RSA_BUFFER);
463  }
464 
465  return 0;
466 }

521 static int RsaPad_OAEP(const byte* input, word32 inputLen, byte* pkcsBlock,
522  word32 pkcsBlockLen, byte padValue, WC_RNG* rng,
523  enum wc_HashType hType, int mgf, byte* optLabel, word32 labelLen,
524  void* heap)
525 {
526  int ret;
527  int hLen;
528  int psLen;
529  int i;
530  word32 idx;
531 
532  byte* dbMask;
533 
534  #ifdef WOLFSSL_SMALL_STACK
535  byte* lHash = NULL;
536  byte* seed = NULL;
537  #else
538  /* must be large enough to contain largest hash */
539  byte lHash[WC_MAX_DIGEST_SIZE];
540  byte seed[ WC_MAX_DIGEST_SIZE];
541  #endif
542 
543  /* no label is allowed, but catch if no label provided and length > 0 */
544  if (optLabel == NULL && labelLen > 0) {
545  return BUFFER_E;
546  }
547 
548  /* limit of label is the same as limit of hash function which is massive */
549  hLen = wc_HashGetDigestSize(hType);
550  if (hLen < 0) {
551  return hLen;
552  }
553 
554  #ifdef WOLFSSL_SMALL_STACK
555  lHash = (byte*)XMALLOC(hLen, heap, DYNAMIC_TYPE_RSA_BUFFER);
556  if (lHash == NULL) {
557  return MEMORY_E;
558  }
559  seed = (byte*)XMALLOC(hLen, heap, DYNAMIC_TYPE_RSA_BUFFER);
560  if (seed == NULL) {
561  XFREE(lHash, heap, DYNAMIC_TYPE_RSA_BUFFER);
562  return MEMORY_E;
563  }
564  #else
565  /* hLen should never be larger than lHash since size is max digest size,
566  but check before blindly calling wc_Hash */
567  if ((word32)hLen > sizeof(lHash)) {
568  WOLFSSL_MSG("OAEP lHash to small for digest!!");
569  return MEMORY_E;
570  }
571  #endif
572 
573  if ((ret = wc_Hash(hType, optLabel, labelLen, lHash, hLen)) != 0) {
574  WOLFSSL_MSG("OAEP hash type possibly not supported or lHash to small");
575  #ifdef WOLFSSL_SMALL_STACK
576  XFREE(lHash, heap, DYNAMIC_TYPE_RSA_BUFFER);
577  XFREE(seed, heap, DYNAMIC_TYPE_RSA_BUFFER);
578  #endif
579  return ret;
580  }
581 
582  /* handles check of location for idx as well as psLen, cast to int to check
583  for pkcsBlockLen(k) - 2 * hLen - 2 being negative
584  This check is similar to decryption where k > 2 * hLen + 2 as msg
585  size approaches 0. In decryption if k is less than or equal -- then there
586  is no possible room for msg.
587  k = RSA key size
588  hLen = hash digest size -- will always be >= 0 at this point
589  */
590  if ((word32)(2 * hLen + 2) > pkcsBlockLen) {
591  WOLFSSL_MSG("OAEP pad error hash to big for RSA key size");
592  #ifdef WOLFSSL_SMALL_STACK
593  XFREE(lHash, heap, DYNAMIC_TYPE_RSA_BUFFER);
594  XFREE(seed, heap, DYNAMIC_TYPE_RSA_BUFFER);
595  #endif
596  return BAD_FUNC_ARG;
597  }
598 
599  if (inputLen > (pkcsBlockLen - 2 * hLen - 2)) {
600  WOLFSSL_MSG("OAEP pad error message too long");
601  #ifdef WOLFSSL_SMALL_STACK
602  XFREE(lHash, heap, DYNAMIC_TYPE_RSA_BUFFER);
603  XFREE(seed, heap, DYNAMIC_TYPE_RSA_BUFFER);
604  #endif
605  return BAD_FUNC_ARG;
606  }
607 
608  /* concatenate lHash || PS || 0x01 || msg */
609  idx = pkcsBlockLen - 1 - inputLen;
610  psLen = pkcsBlockLen - inputLen - 2 * hLen - 2;
611  if (pkcsBlockLen < inputLen) { /*make sure not writing over end of buffer */
612  #ifdef WOLFSSL_SMALL_STACK
613  XFREE(lHash, heap, DYNAMIC_TYPE_RSA_BUFFER);
614  XFREE(seed, heap, DYNAMIC_TYPE_RSA_BUFFER);
615  #endif
616  return BUFFER_E;
617  }
618  XMEMCPY(pkcsBlock + (pkcsBlockLen - inputLen), input, inputLen);
619  pkcsBlock[idx--] = 0x01; /* PS and M separator */
620  while (psLen > 0 && idx > 0) {
621  pkcsBlock[idx--] = 0x00;
622  psLen--;
623  }
624 
625  idx = idx - hLen + 1;
626  XMEMCPY(pkcsBlock + idx, lHash, hLen);
627 
628  /* generate random seed */
629  if ((ret = wc_RNG_GenerateBlock(rng, seed, hLen)) != 0) {
630  #ifdef WOLFSSL_SMALL_STACK
631  XFREE(lHash, heap, DYNAMIC_TYPE_RSA_BUFFER);
632  XFREE(seed, heap, DYNAMIC_TYPE_RSA_BUFFER);
633  #endif
634  return ret;
635  }
636 
637  /* create maskedDB from dbMask */
638  dbMask = (byte*)XMALLOC(pkcsBlockLen - hLen - 1, heap, DYNAMIC_TYPE_RSA);
639  if (dbMask == NULL) {
640  #ifdef WOLFSSL_SMALL_STACK
641  XFREE(lHash, heap, DYNAMIC_TYPE_RSA_BUFFER);
642  XFREE(seed, heap, DYNAMIC_TYPE_RSA_BUFFER);
643  #endif
644  return MEMORY_E;
645  }
646  XMEMSET(dbMask, 0, pkcsBlockLen - hLen - 1); /* help static analyzer */
647 
648  ret = RsaMGF(mgf, seed, hLen, dbMask, pkcsBlockLen - hLen - 1, heap);
649  if (ret != 0) {
650  XFREE(dbMask, heap, DYNAMIC_TYPE_RSA);
651  #ifdef WOLFSSL_SMALL_STACK
652  XFREE(lHash, heap, DYNAMIC_TYPE_RSA_BUFFER);
653  XFREE(seed, heap, DYNAMIC_TYPE_RSA_BUFFER);
654  #endif
655  return ret;
656  }
657 
658  i = 0;
659  idx = hLen + 1;
660  while (idx < pkcsBlockLen && (word32)i < (pkcsBlockLen - hLen -1)) {
661  pkcsBlock[idx] = dbMask[i++] ^ pkcsBlock[idx];
662  idx++;
663  }
664  XFREE(dbMask, heap, DYNAMIC_TYPE_RSA);
665 
666 
667  /* create maskedSeed from seedMask */
668  idx = 0;
669  pkcsBlock[idx++] = 0x00;

790  XMEMCPY(pkcsBlock+pkcsBlockLen-inputLen, input, inputLen);

861  }

870  return MEMORY_E;
871  }
872  XMEMSET(tmp, 0, pkcsBlockLen);
873 
874  /* find seedMask value */
875  if ((ret = RsaMGF(mgf, (byte*)(pkcsBlock + (hLen + 1)),
876  pkcsBlockLen - hLen - 1, tmp, hLen, heap)) != 0) {
877  XFREE(tmp, heap, DYNAMIC_TYPE_RSA_BUFFER);

879  }

890  return ret;

907  return ret;
