168 /* In order to support zero-copy operation, on the RX side we are using custom pbufs, with the payload pointing to the
169  receive buffer obtained from the driver. When the pbuf is eventually freed, the receive buffer is given back to the driver.
170  On the TX side we are incrementing the reference count on the pbuf and giving its payload storage to the driver. Once we
171  detect the transmission is complete, we are freeing our reference to the pbuf. */

209 static err_t enetif_low_level_output(struct netif *netif, struct pbuf *p)
210 {
211  struct pbuf *q;
212  enet_buffer_t bd;
213  status_t status;
214  err_t pbuf_status = ERR_OK;
215  uint8_t pbuf_chain_type = ENETIF_SINGLE_PBUF;
216 
217  LWIP_ASSERT("Output packet buffer empty", p);
218 #if defined(LWIP_DEBUG) && LWIP_NETIF_TX_SINGLE_PBUF && !(LWIP_IPV4 && IP_FRAG) && (LWIP_IPV6 && LWIP_IPV6_FRAG)
219  LWIP_ASSERT("p->next == NULL && p->len == p->tot_len", p->next == NULL && p->len == p->tot_len);
220 #endif
221 
222  /* Check whether this was single or a chained pbuf */
223  if(NULL != p->next)
224  {
225  /* This is a chained pbuf, save info into a local variable, as p will be lost if allocation does not fail */
226  pbuf_chain_type = ENETIF_CHAINED_PBUF;
227  }
228 
229  /* Increment our reference on p */
230  pbuf_ref(p);
231  /* If p was a pbuf chain instead, p's ref was decreased and we got another q pbuf with ref 1
232  Either way, q has a +1 ref that we need to free in case we're not keeping the buffer - ie in case of errors*/
233  q = pbuf_coalesce(p, PBUF_RAW);
234 
235  /* If this was a chained pbuf, check allocation */
236  if((ENETIF_CHAINED_PBUF == pbuf_chain_type) && (q == p))
237  {
238  /* Memory allocation failed */
239  pbuf_status = ERR_MEM;
240  }
241  else
242  {
243  bd.data = q->payload;
244  bd.length = q->tot_len;
245 
246  while(1)
247  {
248  while(OSIF_MutexLock(&enetif_tx_lock, 5) != STATUS_SUCCESS) {}
249  status = ENET_DRV_SendFrame(netif->num, ENET_QUEUE, &bd, NULL);
250  (void)OSIF_MutexUnlock(&enetif_tx_lock);
251 
252  /* Keep trying to send the frame as long as the driver says there is not enough space in the queue */
253  if(status != STATUS_ENET_TX_QUEUE_FULL)
254  {
255  break;
256  }
257  OSIF_TimeDelay(0);
258  }
259  if(status == STATUS_SUCCESS)
260  {
261  /* Post the pbuf to mbox, so that it will be checked for completion in enetif_poll_thread*/
262  sys_mbox_post((sys_mbox_t *)&in_flight_tx_pbufs, (void *)q);
263  }
264  else
265  {
266  /* Decrement the ref (either p's ref in case it was a single pbuf, or the coalesed q's ref) */
267  (void)pbuf_free(q);
268  }
269  pbuf_status = ERR_OK;
270  }
271 
272  return pbuf_status;
273 }

287 {

298  if(netif->input(p, netif) != ERR_OK)
299  {
300  LWIP_DEBUGF(NETIF_DEBUG, ("enetif_input: IP input error\n"));
301  (void)pbuf_free(p);
302  }
303  }
304  return ret;

315 static void enetif_poll_thread(void *arg)
316 {
317  uint8_t instance = (uintptr_t)arg;
318  struct pbuf *p;
319  enet_buffer_t bd;
320  err_t ret;
321  /* Check input parameter */
322  LWIP_ASSERT("g_netif[instance] != NULL", g_netif[instance] != NULL);
323 
324  while (1)
325  {
326  /* Free any completed receive buffers */
327  while (0 == sys_arch_mbox_tryfetch((sys_mbox_t *)&rx_buffs, (void**)&bd.data))
328  {
329  ENET_DRV_ProvideRxBuff(instance, ENET_QUEUE, &bd);
330  }
331 
332  /* Check if there are any new RX frames and provide them to lwip stack */
333  while (STATUS_SUCCESS == ENET_DRV_ReadFrame(instance, ENET_QUEUE, &bd, NULL))
334  {
335  if((NULL == rx_buff_process_handler) || (FORWARD_FRAME != rx_buff_process_handler(instance, &bd)))
336  {
337  ret = enetif_input((struct netif *)g_netif[instance], (uint8_t*)bd.data, (uint16_t)bd.length);
338  if(ERR_OK != ret)
339  {
340  ENET_DRV_ProvideRxBuff(instance, ENET_QUEUE, &bd);
341  }
342  }
343  }
344 
345  /* Check if transmission is complete for any in-flight pbufs */
346  while (0 == sys_arch_mbox_tryfetch((sys_mbox_t *)&in_flight_tx_pbufs, (void**)&p))
347  {
348  if(STATUS_BUSY == ENET_DRV_GetTransmitStatus(instance, ENET_QUEUE, p->payload, NULL))
349  {
350  /* transmission still in progress for the first buffer, repost to (front of) queue and give up for now */
351  sys_mbox_post_to_front((sys_mbox_t *)&in_flight_tx_pbufs, (void *)p);
352  break;
353  }
354  else
355  {
356  /* request to free the outstanding pbuf on tcpip thread */
357  (void)pbuf_free_callback(p);
358  }
359  }
360 
361  OSIF_TimeDelay(0);
362  }
363 }

499 static void enetif_low_level_init(struct netif *netif)
500 {
501  enet_config_t enetConfig;
502  uint8_t i;
503 
504  /* Use only one ring */
505  enet_buffer_config_t enetBufferConfigs[ENET_QUEUE + 1];
506 
507  /* set MAC hardware address length */
508  netif->hwaddr_len = NETIF_MAX_HWADDR_LEN;
509 
510  /* set MAC hardware address */
511  for (i = 0; i < NETIF_MAX_HWADDR_LEN; i++)
512  {
513  netif->hwaddr[i] = netif_cfg[netif->num]->hwaddr[i];
514  }
515 
516  /* maximum transfer unit */
517  netif->mtu = 1500;
518 
519  /* device capabilities */
520  /* don't set NETIF_FLAG_ETHARP if this device is not an ethernet one */
521  netif->flags = (u8_t)(NETIF_FLAG_BROADCAST | NETIF_FLAG_ETHARP | NETIF_FLAG_ETHERNET);
522 #if LWIP_IGMP
523  netif->flags = netif->flags | (u8_t)NETIF_FLAG_IGMP;
524  /*Will add the function igmp_mac_filter to the netif*/
525  (netif)->igmp_mac_filter = igmp_enet_filter;
526 #endif /*LWIP_IGMP*/
527 
528  ENET_DRV_GetDefaultConfig(&enetConfig);
529 
530  enetConfig.miiMode = ENET_MIIMODE;
531  enetConfig.miiSpeed = ENET_MIISPEED;
532 
533 #if FEATURE_ENET_RX_CONFIG_PAYLOAD_LEN_CHECK
534  enetConfig.rxConfig |= (uint32_t)(ENET_RX_CONFIG_ENABLE_PAYLOAD_LEN_CHECK);
535 #endif /* FEATURE_ENET_RX_CONFIG_PAYLOAD_LEN_CHECK */
536 
537 #if FEATURE_ENET_RX_CONFIG_STRIP_CRC_FIELD
538  enetConfig.rxConfig |= (uint32_t)(ENET_RX_CONFIG_STRIP_CRC_FIELD);
539 #endif /* FEATURE_ENET_RX_CONFIG_STRIP_CRC_FIELD */
540 
541 #if FEATURE_ENET_RX_CONFIG_REMOVE_PADDING
542  enetConfig.rxConfig |= (uint32_t)(ENET_RX_CONFIG_REMOVE_PADDING);
543 #endif /* FEATURE_ENET_RX_CONFIG_REMOVE_PADDING */
544 
545  enetConfig.rxConfig |= (uint32_t)(ENET_RX_CONFIG_ENABLE_FLOW_CONTROL);
546 
547 #if FEATURE_ENET_HAS_TX_CONFIG
548  enetConfig.txConfig = 0;
549 #endif /* FEATURE_ENET_HAS_TX_CONFIG */
550 
551 #if FEATURE_ENET_HAS_ACCELERATOR
552  enetConfig.rxAccelerConfig = (uint8_t)(ENET_RX_ACCEL_ENABLE_IP_CHECK | ENET_RX_ACCEL_ENABLE_PROTO_CHECK | ENET_RX_ACCEL_REMOVE_PAD);
553  enetConfig.txAccelerConfig = (uint8_t)(ENET_TX_ACCEL_INSERT_IP_CHECKSUM | ENET_TX_ACCEL_INSERT_PROTO_CHECKSUM);
554 #endif /* FEATURE_ENET_HAS_ACCELERATOR */
555 
556  enetConfig.ringCount = sizeof(enetBufferConfigs) / sizeof (enet_buffer_config_t);
557  enetBufferConfigs[ENET_QUEUE].rxRingSize = ENET_RXBD_NUM;
558  enetBufferConfigs[ENET_QUEUE].txRingSize = ENET_TXBD_NUM;
559  enetBufferConfigs[ENET_QUEUE].rxRingAligned = rxDescriptor;
560  enetBufferConfigs[ENET_QUEUE].txRingAligned = txDescriptor;
561  enetBufferConfigs[ENET_QUEUE].rxBufferAligned = rxBuffer;
562 
563  NETIF_SET_CHECKSUM_CTRL(netif, NETIF_CHECKSUM_SETTING);
564 
565  g_netif[netif->num] = netif;
566  ENET_DRV_Init(netif->num, &enetState, &enetConfig, enetBufferConfigs, netif->hwaddr);
567  ENET_DRV_EnableMDIO(netif->num, false);
568 
569  /* Restart negotiation to speed 10/100M, applied only for PHY_KSZ9031 */
570  if ((enetConfig.miiSpeed == ENET_MII_SPEED_10M) || (enetConfig.miiSpeed == ENET_MII_SPEED_100M))
571  {
572 #if defined(PHY_KSZ9031_SPEED10_100)
573  /* Set 10/100Mbit/s operation and restart negotiation */
574  uint16_t reg;
575  ENET_DRV_MDIORead(netif->num, 7, 0, &reg, 10);
576  reg &= ~(1 << 6);
577  ENET_DRV_MDIOWrite(netif->num, 7, 0, reg, 10);
578  ENET_DRV_MDIORead(netif->num, 7, 9, &reg, 10);
579  reg &= ~((1 << 9) | (1 << 8));
580  ENET_DRV_MDIOWrite(netif->num, 7, 9, reg, 10);
581  ENET_DRV_MDIORead(netif->num, 7, 0, &reg, 10);
582  reg |= (1 << 9);
583  ENET_DRV_MDIOWrite(netif->num, 7, 0, reg, 10);
584 #endif /* defined(PHY_KSZ9031_SPEED10_100) */
585  }
586 
587 #if LWIP_IPV6 && LWIP_IPV6_MLD
588  if(netif_cfg[netif->num]->has_IPv6)
589  {
590  netif->flags = netif->flags | (u8_t)NETIF_FLAG_MLD6;
591  /*If flag MLD6 flag is set, add mdl_enet_filter function to netif*/
592  (netif)->mld_mac_filter = mld_enet_filter;
593  /*
594  * For hardware/netifs that implement MAC filtering.
595  * All-nodes link-local is handled by default, so we must let the hardware know
596  * to allow multicast packets in.
597  * Should set mld_mac_filter previously. */
598  ip6_addr_t ip6_allnodes_ll;
599  ip6_addr_set_allnodes_linklocal(&ip6_allnodes_ll);
600  (void)netif->mld_mac_filter(netif, &ip6_allnodes_ll, NETIF_ADD_MAC_FILTER);
601  }
602 #endif /* LWIP_IPV6 && LWIP_IPV6_MLD */
603 
604  netif_set_link_up(netif);
605 }

620 err_t enet_ethernetif_init(struct netif *netif)
621 {
622  LWIP_ASSERT("netif != NULL", (netif != NULL));
623 
624 #if (NO_SYS == 0)
625  LWIP_MEMPOOL_INIT(RX_POOL);
626  (void)sys_mbox_new((sys_mbox_t *)&rx_buffs, ENET_RXBD_NUM);
627  (void)sys_mbox_new((sys_mbox_t *)&in_flight_tx_pbufs, 2 * ENET_TXBD_NUM);
628  status_t status = OSIF_MutexCreate(&enetif_tx_lock);
629 
630  LWIP_ASSERT("status == STATUS_SUCCESS", status == STATUS_SUCCESS);
631  (void)status;
632 #endif /* (NO_SYS == 0) */
633 
634  netif->name[0] = netif_cfg[netif->num]->name[0];
635  netif->name[1] = netif_cfg[netif->num]->name[1];
636 
637 #if LWIP_IPV4
638 #if LWIP_ARP
639  /* We directly use etharp_output() here to save a function call.
640  * You can instead declare your own function an call etharp_output()
641  * from it if you have to do some checks before sending (e.g. if link
642  * is available...) */
643  netif->output = etharp_output;
644 #else /* LWIP_ARP */
645  netif->output = NULL; /* not used for PPPoE */
646 #endif /* LWIP_ARP */
647 #endif /* LWIP_IPV4 */
648 #if LWIP_IPV6
649  if(netif_cfg[netif->num]->has_IPv6)
650  {
651  netif->output_ip6 = ethip6_output;
652  }
653 #endif /* LWIP_IPV6 */
654  netif->linkoutput = enetif_low_level_output;
655 #if LWIP_NETIF_HOSTNAME
656  /* Initialize interface hostname */
657  if(NULL != netif_cfg[netif->num]->hostname)
658  {
659  netif->hostname = netif_cfg[netif->num]->hostname;
660  }
661 #endif /* LWIP_NETIF_HOSTNAME */
662 #if LWIP_SNMP
663  /*
664  * Initialize the snmp variables and counters inside the struct netif.
665  * The last argument should be replaced with your link speed, in units
666  * of bits per second.
667  */
668  NETIF_INIT_SNMP(netif, (u8_t)snmp_ifType_ethernet_csmacd, (u32_t)100000000);
669 #endif /* LWIP_SNMP */
670 
671  /* initialize the hardware */
672  enetif_low_level_init(netif);
673 
674 #if (NO_SYS == 0)
675  /* Start the polling thread */
676  poll_thread = sys_thread_new("enetif_poll_thread", enetif_poll_thread, (void*)((uint32_t)netif->num), DEFAULT_THREAD_STACKSIZE, DEFAULT_THREAD_PRIO);
677 #endif /* (NO_SYS == 0) */
678 
679  return ERR_OK;
680 }

688 void enet_ethernetif_shutdown(struct netif *netif)
689 {
690 #if (NO_SYS == 0)
691  struct pbuf *p;
692  uint8_t *ptr;
693 
694  LWIP_ASSERT("netif != NULL", (netif != NULL));
695 
696  /* Kill the polling thread */
697  sys_thread_delete(poll_thread);
698 
699  /* Empty and free the mboxes */
700  while (0 == sys_arch_mbox_tryfetch((sys_mbox_t *)&in_flight_tx_pbufs, (void**)&p))
701  {
702  (void)pbuf_free_callback(p);
703  }
