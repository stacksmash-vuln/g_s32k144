97 STATIC INLINE word32 ByteReverseWord32(word32 value)
98 {
99 #ifdef PPC_INTRINSICS
100  /* PPC: load reverse indexed instruction */
101  return (word32)__lwbrx(&value,0);
102 #elif defined(__ICCARM__)
103  return (word32)__REV(value);
104 #elif defined(KEIL_INTRINSICS)
105  return (word32)__rev(value);
106 #elif defined(FAST_ROTATE)
107  /* 5 instructions with rotate instruction, 9 without */
108  return (rotrFixed(value, 8U) & 0xff00ff00) |
109  (rotlFixed(value, 8U) & 0x00ff00ff);
110 #elif defined(NXP_SDK)
111  register word32 r;
112  REV_BYTES_32(value, r);
113  return r;
114 #else
115  /* 6 instructions with rotate instruction, 8 without */
116  value = ((value & 0xFF00FF00) >> 8) | ((value & 0x00FF00FF) << 8);
117  return rotlFixed(value, 16U);
118 #endif
119 }
