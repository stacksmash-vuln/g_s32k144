145 static bool FLASH_MX25L6433F_DRV_Timeout(uint32_t startTime, uint32_t timeout)
146 {
147  uint32_t currentTime;
148  bool retVal;
149 
150  currentTime = OSIF_GetMilliseconds();
151  if (currentTime >= startTime)
152  {
153  retVal = ((currentTime - startTime) > timeout)?true:false;
154  }
155  else
156  {
157  /* wrap around */
158  retVal = ((FLASH_MX25L6433F_TIMEOUT_WRAP - startTime + currentTime) > timeout)?true:false;
159  }
160  return retVal;
161 }

164 /*FUNCTION**********************************************************************
165  *
166  * Function Name : FLASH_MX25L6433F_DRV_WriteEnable
167  * Description : Enables the serial flash memory for a program or erase operation
168  *
169  *END**************************************************************************/
170 static status_t FLASH_MX25L6433F_DRV_WriteEnable(uint32_t instance)
171 {
172  uint32_t retries = FLASH_MX25L6433F_MAX_RETRY + 1U;
173  status_t status = STATUS_TIMEOUT;
174  status_t cmdStatus;
175  uint8_t srValue = 0U;
176 
177  while (retries > 0U)
178  {
179  /* send WREN command */
180  cmdStatus = QSPI_DRV_IpCommand(instance, FLASH_MX25L6433F_LUT_WREN, 1U);
181  /* send RDSR command */
182  if (cmdStatus == STATUS_SUCCESS)
183  {
184  cmdStatus = QSPI_DRV_IpRead(instance, FLASH_MX25L6433F_LUT_RDSR, 0U, &srValue, NULL, 1U, QSPI_TRANSFER_TYPE_SYNC, FLASH_MX25L6433F_TIMEOUT);
185  }
186  if (cmdStatus == STATUS_SUCCESS)
187  {
188  /* check WEL == 1 */
189  if ((srValue & FLASH_MX25L6433F_SR_WEL_MASK) != 0U)
190  {
191  status = STATUS_SUCCESS;
192  break;
193  }
194  }
195  else
196  {
197  /* record error */
198  status = cmdStatus;
199  }
200  retries--;
201  }
202 
203  return status;
204 }

208  *
209  * Function Name : FLASH_MX25L6433F_DRV_CheckLastCommand
210  * Description : Checks the outcome of the last program/erase operation
211  *
212  *END**************************************************************************/
213 static status_t FLASH_MX25L6433F_DRV_CheckLastCommand(uint32_t instance)
214 {
215  uint8_t secValue;
216  status_t status = STATUS_ERROR;
217  const flash_mx25l6433f_state_t * state;
218 
219  state = g_flashMx25l6433fStatePtr[instance];
220 
221  switch (state->lastCommand)
222  {
223  case FLASH_MX25L6433F_LASTCMD_WRITE:
224  /* check success of program command */
225  /* read security register */
226  status = QSPI_DRV_IpRead(instance, FLASH_MX25L6433F_LUT_RDSCUR, 0U, &secValue, NULL, 1U, QSPI_TRANSFER_TYPE_SYNC, FLASH_MX25L6433F_TIMEOUT);
227  if ((status == STATUS_SUCCESS) && ((secValue & FLASH_MX25L6433F_SEC_P_FAIL_MASK) != 0U))
228  {
229  status = STATUS_ERROR;
230  }
231  else
232  {
233  status = STATUS_SUCCESS;
234  }
235  break;
236  case FLASH_MX25L6433F_LASTCMD_ERASE:
237  /* check success of erase command */
238  /* read security register */
239  status = QSPI_DRV_IpRead(instance, FLASH_MX25L6433F_LUT_RDSCUR, 0U, &secValue, NULL, 1U, QSPI_TRANSFER_TYPE_SYNC, FLASH_MX25L6433F_TIMEOUT);
240  if ((status == STATUS_SUCCESS) && ((secValue & FLASH_MX25L6433F_SEC_E_FAIL_MASK) != 0U))
241  {
242  status = STATUS_ERROR;
243  }
244  else
245  {
246  status = STATUS_SUCCESS;
247  }
248  break;
249  case FLASH_MX25L6433F_LASTCMD_OTHER:
250  default:
251  status = STATUS_SUCCESS;
252  break;
253  }
254  return status;

384 /*FUNCTION**********************************************************************
385  *
386  * Function Name : FLASH_MX25L6433F_DRV_InitDevice
387  * Description : Initializes the MX25L6433F device
388  *
389  *END**************************************************************************/
390 static inline status_t FLASH_MX25L6433F_DRV_InitDevice(uint32_t instance,
391  const flash_mx25l6433f_user_config_t * userConfigPtr)
392 {
393  uint8_t srValue;
394  uint8_t crValue;
395  uint8_t srWrite[2U];
396  status_t status;
397  uint32_t startTime;
398 
399  /* Enable quad operations, configure drive strength, configure dummy cycles to 0 (<80MHz) */
400  /* send RDSR command */
401  status = QSPI_DRV_IpRead(instance, FLASH_MX25L6433F_LUT_RDSR, 0U, &srValue, NULL, 1U, QSPI_TRANSFER_TYPE_SYNC, FLASH_MX25L6433F_TIMEOUT);
402  if (status != STATUS_SUCCESS)
403  {
404  return status;
405  }
406  /* send RDCR command */
407  status = QSPI_DRV_IpRead(instance, FLASH_MX25L6433F_LUT_RDCR, 0U, &crValue, NULL, 1U, QSPI_TRANSFER_TYPE_SYNC, FLASH_MX25L6433F_TIMEOUT);
408  if (status != STATUS_SUCCESS)
409  {
410  return status;
411  }
412  /* Check if current settings are ok */
413  if (((srValue & FLASH_MX25L6433F_SR_QE_MASK) == 0U) ||
414  (((crValue & FLASH_MX25L6433F_CFG_ODS_MASK) >> FLASH_MX25L6433F_CFG_ODS_SHIFT) != (uint8_t)userConfigPtr->outputDriverStrength) ||
415  ((crValue & FLASH_MX25L6433F_CFG_DC_MASK) != 0U))
416  {
417  /* enable write before WRSR command */
418  status = FLASH_MX25L6433F_DRV_WriteEnable(instance);
419  if (status != STATUS_SUCCESS)
420  {
421  return status;
422  }
423  /* QE not set, write SR to set it */
424  srWrite[0U] = (uint8_t)(srValue | FLASH_MX25L6433F_SR_QE_MASK);
425  crValue &= (uint8_t)(~(FLASH_MX25L6433F_CFG_ODS_MASK | FLASH_MX25L6433F_CFG_DC_MASK));
426  srWrite[1U] = (uint8_t)(crValue | FLASH_MX25L6433F_CFG_ODS(userConfigPtr->outputDriverStrength));
427  status = QSPI_DRV_IpWrite(instance, FLASH_MX25L6433F_LUT_WRSR, 0U, srWrite, 2U, QSPI_TRANSFER_TYPE_SYNC, FLASH_MX25L6433F_TIMEOUT);
428  if (status != STATUS_SUCCESS)
429  {
430  return status;
431  }
432  /* Wait for the command to complete. */
433  startTime = OSIF_GetMilliseconds();
434  do
435  {
436  status = FLASH_MX25L6433F_DRV_GetStatus(instance);
437  }
438  while (!FLASH_MX25L6433F_DRV_Timeout(startTime, FLASH_MX25L6433F_TIMEOUT) && (status == STATUS_BUSY));
439  if (status == STATUS_BUSY)
440  {
441  status = STATUS_TIMEOUT;
442  }
443  }
444  return status;
445 }

460 status_t FLASH_MX25L6433F_DRV_Init(uint32_t instance,
461  const flash_mx25l6433f_user_config_t * userConfigPtr,
462  flash_mx25l6433f_state_t * state)
463 {
464  status_t status;
465 
466  DEV_ASSERT(instance < QuadSPI_INSTANCE_COUNT);
467  DEV_ASSERT(g_flashMx25l6433fStatePtr[instance] == NULL);
468 
469  /* Copy configuration parameters to state structure */
470  state->outputDriverStrength = userConfigPtr->outputDriverStrength;
471  state->dmaSupport = userConfigPtr->dmaSupport;
472  state->lastCommand = (uint8_t)FLASH_MX25L6433F_LASTCMD_NONE;
473  state->suspendedCommand = (uint8_t)FLASH_MX25L6433F_LASTCMD_NONE;
474  /* QuadSPI is assumed to be already initialized by the application */
475  /* Initialize LUT */
476  FLASH_MX25L6433F_DRV_InitLut(instance);
477  /* AHB Setup */
478  QSPI_DRV_SetAhbSeqId(instance, FLASH_MX25L6433F_LUT_4READ);
479 
480  /* Initialize device */
481  status = FLASH_MX25L6433F_DRV_InitDevice(instance, userConfigPtr);
