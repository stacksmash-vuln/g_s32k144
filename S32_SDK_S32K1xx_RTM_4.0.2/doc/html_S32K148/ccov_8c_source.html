157 static void
158 Coverage_thread(void *arg)
159 {
160  struct netconn *conn;
161  struct netbuf *buf;
162  TaskHandle_t TaskHandler = NULL;
163  err_t err;
164  LWIP_UNUSED_ARG(arg);
165  ip_addr_t * remote_computer_addr;
166  uint16_t * remote_computer_port;
167 
168 #if LWIP_IPV6
169  conn = netconn_new(NETCONN_UDP_IPV6);
170  netconn_bind(conn, IP6_ADDR_ANY, UDP_MESSAGE_SEND_PORT);
171 #else /* LWIP_IPV6 */
172  conn = netconn_new(NETCONN_UDP);
173  netconn_bind(conn, IP_ADDR_ANY, UDP_MESSAGE_SEND_PORT);
174 #endif /* LWIP_IPV6 */
175 
176  LWIP_ERROR("udpecho: invalid conn", (conn != NULL), return;);
177 
178  for(;;)
179  {
180  if ((TaskHandler!= NULL)&&(Coverage_test_done_flag == 1))
181  {
182  vTaskDelete(TaskHandler);
183  TaskHandler = NULL;
184  Coverage_test_done_flag = 0;
185  }
186  /*Polling for msg from connection*/
187  err = netconn_recv(conn, &buf);
188  if (err == ERR_OK)
189  {
190  if(netbuf_copy(buf, buffer1, sizeof(buffer1)) != buf->p->tot_len)
191  {
192  LWIP_DEBUGF(LWIP_DBG_ON, ("netbuf_copy failed\n"));
193  }
194  /* no need netconn_connect here, since the netbuf contains the address */
195  else
196  {
197  /* if receive message "COVERAGE_TEST", start to overload the mailbox */
198  if (buf->p->tot_len != 0 && !strncmp(&buffer1[0], "COVERAGE_TEST", 13))
199  {
200  /*start thread*/
201  ETHIF_REGISTER_RX_BUFF_PROCESS_CONDITION_HANDLER(handler);
202  BaseType_t ret =xTaskCreate(CoverageSendTask, "SendTask", 256U, NULL, DEFAULT_THREAD_PRIO, &TaskHandler);
203  LWIP_ASSERT("failed to create SendTask", ret == pdPASS);
204  Coverage_test_done_flag = 0;
205  /*get the ip and port of the computer for future use*/
206  netconn_getaddr(conn, &remote_computer_addr, &remote_computer_port, 0);
207  }
208  /* if receive message "NETIF_SHUTDOWN", call function ETHIF_SHUTDOWN() with the
209  netif used for this echo server */
210  if (buf->p->tot_len == 14 && !strncmp(&buffer1[0], "NETIF_SHUTDOWN", 14))
211  {
212  send_tx_pbuffs_dummy_char();
213  send_rx_pbuffs_dummy_char();
214  ETHIF_REGISTER_RX_BUFF_PROCESS_CONDITION_HANDLER(NULL);
215  ETHIF_SHUTDOWN(netif_shutdown);
216  }
217  netbuf_delete(buf);
218  }
219  }
220  }
221 }
