298 /* This function initializes all network interfaces
299  * Implements enetif_init_Activity
300  */
301 static void interface_init(void)
302 {
303  for(int i = 0; i < ETHIF_NUMBER; i++)
304  {
305 #if LWIP_IPV4
306  ip4_addr_t ipaddr, netmask, gw;
307 #endif /* LWIP_IPV4 */
308 #if LWIP_DHCP || LWIP_AUTOIP
309  err_t err;
310 #endif /* LWIP_AUTOIP || LWIP_DHCP */
311 
312 #if LWIP_IPV4
313  ip4_addr_set_zero(&gw);
314  ip4_addr_set_zero(&ipaddr);
315  ip4_addr_set_zero(&netmask);
316  /* network_interfaces[i] takes the IPV4 addresses from the respective configuration */
317  if ((!netif_cfg[i]->has_dhcp) && (!netif_cfg[i]->has_auto_ip))
318  {
319  IP4_ADDR((&gw), netif_cfg[i]->gw[0], netif_cfg[i]->gw[1], netif_cfg[i]->gw[2], netif_cfg[i]->gw[3]);
320  IP4_ADDR((&ipaddr), netif_cfg[i]->ip_addr[0], netif_cfg[i]->ip_addr[1], netif_cfg[i]->ip_addr[2], netif_cfg[i]->ip_addr[3]);
321  IP4_ADDR((&netmask), netif_cfg[i]->netmask[0], netif_cfg[i]->netmask[1], netif_cfg[i]->netmask[2], netif_cfg[i]->netmask[3]);
322  }
323 #endif /* LWIP_IPV4 */
324 
325 #if NO_SYS
326  netif_set_default(netif_add(&network_interfaces[i], &ipaddr, &netmask, &gw, NULL, ETHIF_INIT, netif_input));
327 #else /* NO_SYS */
328  netif_set_default(netif_add(&network_interfaces[i], &ipaddr, &netmask, &gw, NULL, ETHIF_INIT, tcpip_input));
329 #endif /* NO_SYS */
330 
331 #if LWIP_IPV6
332  if (netif_cfg[i]->has_IPv6)
333  {
334  netif_create_ip6_linklocal_address(&network_interfaces[i], 1);
335 
336 #if PRINTF_SUPPORT
337  printf("ip6 linklocal address: ");
338 #endif
339  ip6_addr_debug_print(0xFFFFFFFFU & ~LWIP_DBG_HALT, netif_ip6_addr(&network_interfaces[i], 0));
340  }
341 #endif /* LWIP_IPV6 */
342 
343 #if LWIP_NETIF_STATUS_CALLBACK
344  netif_set_status_callback(&network_interfaces[i], status_callback);
345 #endif /* LWIP_NETIF_STATUS_CALLBACK */
346 
347 #if LWIP_NETIF_LINK_CALLBACK
348  netif_set_link_callback(&network_interfaces[i], link_callback);
349 #endif /* LWIP_NETIF_LINK_CALLBACK */
350 
351 #if LWIP_AUTOIP
352  if (netif_cfg[i]->has_auto_ip)
353  {
354  autoip_set_struct(&network_interfaces[i], &netif_autoip);
355  }
356 #endif /* LWIP_AUTOIP */
357 
358 #if LWIP_DHCP
359  if (netif_cfg[i]->has_dhcp)
360  {
361  dhcp_set_struct(&network_interfaces[i], &netif_dhcp);
362  }
363 #endif /* LWIP_DHCP */
364  netif_set_up(&network_interfaces[i]);
365 #if LWIP_DHCP
366  if (netif_cfg[i]->has_dhcp)
367  {
368  err = dhcp_start((struct netif *)&network_interfaces[i]);
369  LWIP_ASSERT("dhcp_start failed", err == (err_t)ERR_OK);
370  }
371 #endif /* LWIP_DHCP */
372 
373 #if LWIP_AUTOIP
374  else if (netif_cfg[i]->has_auto_ip)
375  {
376  err = autoip_start(&network_interfaces[i]);
377  LWIP_ASSERT("autoip_start failed", err == (err_t)ERR_OK);
378  }
379 #endif /* LWIP_AUTOIP */
380  }
381 }

569 static void mainLoopTask(void* pvParameters)
570 {
571  (void)pvParameters;
572 
573  /* initialize lwIP stack, network interfaces and applications */
574 #if !NO_SYS
575  err_t err;
576  sys_sem_t init_sem;
577 
578 #if defined (CPU_MPC5777C)|| defined (CPU_MPC5746R)
579 #ifdef PIT_0
580 #define PIT PIT_0
581 #endif
582  PIT->MCR &= (uint32_t)(~PIT_MCR_FRZ_MASK);/* stop the timer in debug workaround for compability on MPC5777C PIT timer*/
583 #endif
584 
585  err = sys_sem_new(&init_sem, 0);
586  LWIP_ASSERT("failed to create init_sem", err == (err_t)ERR_OK);
587  LWIP_UNUSED_ARG(err);
588  tcpip_init(test_init, (void*)&init_sem);
589  /* we have to wait for initialization to finish before
590  * calling update_adapter()! */
591  (void)sys_sem_wait(&init_sem);
592  sys_sem_free(&init_sem);
593 #if (LWIP_SOCKET || LWIP_NETCONN) && LWIP_NETCONN_SEM_PER_THREAD
594  netconn_thread_init();
595 #endif
596 #else /* !NO_SYS */
597  sys_init();
598  lwip_init();
599  test_init(NULL);
600 #endif /* !NO_SYS */
601 
602 
603 #if LWIP_INIT_COMPLETE_CALLBACK
604  tcpip_init_complete_callback();
605 #endif /* LWIP_INIT_COMPLETE_CALLBACK */
606 
607  /* MAIN LOOP for driver update (and timers if NO_SYS) */
608  while (1) {
609 #if NO_SYS
610  /* handle timers (already done in tcpip.c when NO_SYS=0) */
611  sys_check_timeouts();
612  for(int i = 0; i < ETHIF_NUMBER ; i++)
613  {
614  (void)ETHIF_POLL_INTERFACE(&network_interfaces[i]);
615  }
616 #else /* NO_SYS */
617  sys_msleep(5000);
618 #endif /* NO_SYS */
619  }
620 
621 #if (LWIP_SOCKET || LWIP_NETCONN) && LWIP_NETCONN_SEM_PER_THREAD
622  netconn_thread_cleanup();
623 #endif
624  /* release the network interfaces... */
625  for(int i = 0 ; i < ETHIF_NUMBER ; i++)
626  {
627  ETHIF_SHUTDOWN(&network_interfaces[i]);
628  }
629 }

674 void vApplicationMallocFailedHook(void)
675 {
676  /* vApplicationMallocFailedHook() will only be called if
677  configUSE_MALLOC_FAILED_HOOK is set to 1 in FreeRTOSConfig.h. It is a hook
678  function that will get called if a call to pvPortMalloc() fails.
