119 /*FUNCTION**********************************************************************
120  *
121  * Function Name : RTC_DRV_IsYearLeap
122  * Description : This method checks if the year passed as a parameter is a leap
123  * one.
124  * Return : True if the year is leap, false if otherwise.
125  * Implements : RTC_DRV_IsYearLeap_Activity
126  *END**************************************************************************/
127 static bool RTC_DRV_IsYearLeap(uint16_t year)
128 {
129  bool isYearLeap = false;
130 
131  if ((year % 4U) > 0U)
132  {
133  isYearLeap = false;
134  }
135  else if ((year % 100U) > 0U)
136  {
137  isYearLeap = true;
138  }
139  else if ((year % 400U) > 0U)
140  {
141  isYearLeap = false;
142  }
143  else
144  {
145  isYearLeap = true;
146  }
147 
148  /* Return the exit code */
149  return isYearLeap;
150 }

152 /*FUNCTION**********************************************************************
153  *
154  * Function Name : RTC_DRV_ConvertTimeDateToSeconds
155  * Description : This method converts time-date into seconds.
156  * Return : None
157  * Implements : RTC_DRV_ConvertTimeDateToSeconds_Activity
158  *END**************************************************************************/
159 static void RTC_DRV_ConvertTimeDateToSeconds(rtc_timedate_t * timeDate, uint32_t * seconds)
160 {
161  DEV_ASSERT(seconds != NULL);
162  DEV_ASSERT(timeDate != NULL);
163 
164  /* Declare local variables */
165  uint16_t year;
166 
167  /* Convert years to seconds */
168  (*seconds) = (uint32_t)(DAYS_IN_A_YEAR * (uint32_t)(SECONDS_IN_A_DAY));
169  (*seconds) *= ((uint32_t)timeDate->year - YEAR_RANGE_START);
170 
171  /* Add the seconds from the leap years */
172  for (year = YEAR_RANGE_START; year < timeDate->year; year++)
173  {
174  if (RTC_DRV_IsYearLeap(year))
175  {
176  (*seconds) += SECONDS_IN_A_DAY;
177  }
178  }
179 
180  /* If the current year is leap and 29th of February has passed, add
181  * another day to seconds passed.
182  */
183  if ((RTC_DRV_IsYearLeap(year)) && (timeDate->month > 2U))
184  {
185  (*seconds) += SECONDS_IN_A_DAY;
186  }
187 
188  /* Add the rest of the seconds from the current month */
189  (*seconds) += MONTH_DAYS[timeDate->month] * SECONDS_IN_A_DAY;
190  /* Add the rest of the seconds from the current day */
191  (*seconds) += (uint32_t)(((uint32_t)timeDate->day - 1U) * (uint32_t)SECONDS_IN_A_DAY);
192  /* Add the rest of the seconds from the current time */
193  (*seconds) += (uint32_t)(((uint32_t)timeDate->hour * SECONDS_IN_A_HOUR) + \
194  ((uint32_t)timeDate->minutes * SECONDS_IN_A_MIN) + \
195  (uint32_t)timeDate->seconds);
196 }

201 int nxp_sdk_port_init(void)
202 {
203  status_t ret = STATUS_SUCCESS;
204 
205 #if defined(NXP_SDK_HSM)
206  ret = HSM_DRV_Init(&g_hsm_state);
207  DEV_ASSERT(ret == STATUS_SUCCESS);
208  ret = HSM_DRV_InitRNG(HSM_TIMEOUT);
209  DEV_ASSERT(ret == STATUS_SUCCESS);
210 #endif /* NXP_SDK_HSM */
211 
212 #if defined(NXP_SDK_CSEC)
213  CSEC_DRV_Init(&g_csec_state);
214  ret = CSEC_DRV_InitRNG();
215  DEV_ASSERT(ret == STATUS_SUCCESS);
216 #endif /* NXP_SDK_CSEC */
217 
218 #if defined(NXP_SDK_CSE)
219  kia[4] = (((uint32_t)&firmware) & 0xff000000) >> 24;
220  kia[5] = (((uint32_t)&firmware) & 0xff0000) >> 16;
221  kia[6] = (((uint32_t)&firmware) & 0xff00) >> 8;
222  kia[7] = (((uint32_t)&firmware) & 0xff);
223  CSE_DRV_SetKeyImageAddress((uint32_t)&kia, (uint32_t)&kia);
224  CSE_DRV_Init(&g_cse_state);
225  ret = CSE_DRV_InitRNG(10);
226  DEV_ASSERT(ret == STATUS_SUCCESS);
227 #endif /* NXP_SDK_CSE */
228 
229 #if defined(NXP_SDK_RTC_C55) || defined(NXP_SDK_RTC_S32K)
230  rtc_init_config_t rtcConfig;
231  RTC_DRV_GetDefaultConfig(&rtcConfig);
232 #if defined(NXP_SDK_RTC_C55)
233  RTC_DRV_Init(RTC_INSTANCE, &g_rtc_state, &rtcConfig);
234 #elif defined(NXP_SDK_RTC_S32K)
235  RTC_DRV_Init(RTC_INSTANCE, &rtcConfig);
236 #endif /* NXP_SDK_RTC_C55 */
237 #endif /* NXP_SDK_RTC_C55 || NXP_SDK_RTC_S32K */
238 
239 #if (defined(NXP_SDK_RTC_C55) || defined(NXP_SDK_RTC_S32K) || defined (NXP_SDK_PIT))
240  /* Initialize Time & Date based on host system's clock at compile time.
241  This may need to be initialized more accurately.
242  */
243  rtc_timedate_t timedate;
244  char dbuff[] = __DATE__;
245  char tbuff[] = __TIME__;
246 
247  timedate.year = (uint16_t)atoi((const char*)&dbuff[7]);
248  dbuff[6] = '\0';
249  timedate.day = (uint16_t)atoi((const char*)&dbuff[4]);
250  dbuff[3] = '\0';
251  timedate.month = (uint16_t)((lwip_strnstr(month_names, dbuff, sizeof(month_names)) - month_names) / 3 + 1);
252 
253  timedate.seconds = (uint8_t)atoi((const char*)&tbuff[6]);
254  tbuff[5] = '\0';
255  timedate.minutes = (uint16_t)atoi((const char*)&tbuff[3]);
256  tbuff[2] = '\0';
257  timedate.hour = (uint16_t)atoi((const char*)&tbuff[0]);
258 #endif /* (defined(NXP_SDK_RTC_C55) || defined(NXP_SDK_RTC_S32K) || defined (NXP_SDK_PIT)) */
259 
260 #if (defined(NXP_SDK_RTC_C55) || defined(NXP_SDK_RTC_S32K))
261  ret = RTC_DRV_SetTimeDate(RTC_INSTANCE, &timedate);
262  DEV_ASSERT(ret == STATUS_SUCCESS);
263  ret = RTC_DRV_StartCounter(RTC_INSTANCE);
264  DEV_ASSERT(ret == STATUS_SUCCESS);
265 #elif defined (NXP_SDK_PIT)
266  RTC_DRV_ConvertTimeDateToSeconds(&timedate, &PIT_sec);
267  RTC_DRV_StartCounter(RTC_INSTANCE);
268 #endif /* NXP_SDK_RTC_C55 || NXP_SDK_RTC_S32K */
269  return ret;
270 }
