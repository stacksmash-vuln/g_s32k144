207 /*
208  Blocks the thread until a message arrives in the mailbox, but does
209  not block the thread longer than "timeout" milliseconds (similar to
210  the sys_arch_sem_wait() function). The "msg" argument is a result
211  parameter that is set by the function (i.e., by doing "*msg =
212  ptr"). The "msg" parameter maybe NULL to indicate that the message
213  should be dropped.
214 
215  The return values are the same as for the sys_arch_sem_wait() function:
216  Number of milliseconds spent waiting or SYS_ARCH_TIMEOUT if there was a
217  timeout.
218 
219  Note that a function with a similar name, sys_mbox_fetch(), is
220  implemented by lwIP.
221 */
222 u32_t sys_arch_mbox_fetch(sys_mbox_t *mbox, void **msg, u32_t timeout)
223 {
224  void** msgptr = msg;
225  u32_t startTime;
226 
227  LWIP_ASSERT("mbox != NULL", mbox != NULL);
228 
229  startTime = sys_now();
230 
231  if (timeout != 0U)
232  {
233  if (pdTRUE == xQueueReceive(*mbox, msgptr, pdMS_TO_TICKS(timeout)))
234  {
235  return get_elapsed_time(startTime);
236  }
237  else /* timed out blocking for message */
238  {
239  *msgptr = NULL;
240  return SYS_ARCH_TIMEOUT;
241  }
242  }
243  else /* block forever for a message. */
244  {
245  while (pdTRUE != xQueueReceive(*mbox, msgptr, 10000)) /* time is arbitrary */
246  {
247  ;
248  }
249  return get_elapsed_time(startTime);
250  }
251 }

297 /*-----------------------------------------------------------------------------------*/
298 /*
299  Blocks the thread while waiting for the semaphore to be
300  signaled. If the "timeout" argument is non-zero, the thread should
301  only be blocked for the specified time (measured in
302  milliseconds).
303 
304  If the timeout argument is non-zero, the return value is the number of
305  milliseconds spent waiting for the semaphore to be signaled. If the
306  semaphore wasn't signaled within the specified time, the return value is
307  SYS_ARCH_TIMEOUT. If the thread didn't have to wait for the semaphore
308  (i.e., it was already signaled), the function may return zero.
309 
310  Notice that lwIP implements a function with a similar name,
311  sys_sem_wait(), that uses the sys_arch_sem_wait() function.
312 */
313 u32_t sys_arch_sem_wait(sys_sem_t *sem, u32_t timeout)
314 {
315  u32_t startTime;
316 
317  LWIP_ASSERT("sem != NULL", sem != NULL);
318 
319  startTime = sys_now();
320 
321  if (timeout != 0U)
322  {
323  if (pdTRUE == xSemaphoreTake(*sem, pdMS_TO_TICKS(timeout)))
324  {
325  return get_elapsed_time(startTime);
326  }
327  else
328  {
329  return SYS_ARCH_TIMEOUT;
330  }
331  }
332  else /* must block without a timeout */
333  {
334  while (xSemaphoreTake(*sem, 10000) != pdTRUE)
335  {
336  ;
337  }
338  return get_elapsed_time(startTime);
339  }
340 }

414 /* Utility method that wraps a thread function call and provides a way to add initialization - eg per-thread semaphores
415  * Implements design_id_SOCKETS_Activity
416  */
417 static void sys_thread_function(void* arg)
418 {
419  struct threadfunc* t = (struct threadfunc*)arg;
420 #if (LWIP_SOCKET || LWIP_NETCONN) && LWIP_NETCONN_SEM_PER_THREAD
421  netconn_thread_init();
422 #endif
423  (void)t->function(t->arg);
424  /* Regular FreeRTOS tasks don't return. But in case they do, make them exit cleanly. */
425 #if (LWIP_SOCKET || LWIP_NETCONN) && LWIP_NETCONN_SEM_PER_THREAD
426  netconn_thread_cleanup();
427 #endif
428  mem_free(t);
429  sys_thread_delete(NULL);
430 }

432 /*-----------------------------------------------------------------------------------
433  Starts a new thread named "name" with priority "prio" that will begin its
434  execution in the function "thread()". The "arg" argument will be passed as an
435  argument to the thread() function. The stack size to used for this thread is
436  the "stacksize" parameter. The id of the new thread is returned. Both the id
437  and the priority are system dependent.
438 */
439 sys_thread_t sys_thread_new(const char *name, lwip_thread_fn thread, void *arg, int stacksize, int prio)
440 {
441  TaskHandle_t CreatedTask = NULL;
442  uint16_t stackDepth = ((uint16_t)stacksize != 0U) ? (uint16_t)stacksize : (uint16_t)configMINIMAL_STACK_SIZE;
443  struct threadfunc* tf = (struct threadfunc*)mem_malloc(sizeof (struct threadfunc));
444 
445  LWIP_ASSERT("tf != NULL", tf != NULL);
446 
447  tf->function = thread;
448  tf->arg = arg;
449 
450  xTaskCreate(sys_thread_function, name, stackDepth, tf, prio, &CreatedTask);
451 
452  return CreatedTask;
453 }
