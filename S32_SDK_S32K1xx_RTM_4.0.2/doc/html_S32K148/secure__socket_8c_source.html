151 char* bench_aescbc()
152 {
153  static char result[32] = "Enc: KB/s Dec: KB/s\n";
154  char *p;
155  double start_time, done_time;
156  int i = 0;
157  int ret = 0;
158 
159  memset(bench_plain, 0, BLOCK_SIZE);
160  memset(bench_cipher, 0, BLOCK_SIZE);
161  memset(enc, 0, sizeof(*enc));
162 
163  /* init keys */
164  ret = wc_AesInit(enc, NULL, INVALID_DEVID);
165  LWIP_ASSERT("ws_AesInit() failed", ret == 0);
166  ret = wc_AesSetKey(enc, (byte*)bench_key, 16, (byte*)bench_iv, AES_ENCRYPTION);
167  LWIP_ASSERT("ws_AesSetKey() failed", ret == 0);
168 
169  start_time = current_time();
170  for (i = 0; i < NUM_BLOCKS; i++) {
171  ret = wc_AesCbcEncrypt(enc, bench_plain, bench_cipher, BLOCK_SIZE);
172  LWIP_ASSERT("ws_AesCbcEncrypt() failed", ret == 0);
173  }
174  done_time = current_time();
175  p = &result[4];
176  lwip_itoa(p, 5, SPEED(start_time, done_time));
177  while (*p)
178  p++;
179  *p = ' ';
180 
181  /* init keys */
182  ret = wc_AesSetKey(enc, (byte*)bench_key, 16, (byte*)bench_iv, AES_DECRYPTION);
183  LWIP_ASSERT("ws_AesSetKey() failed", ret == 0);
184  start_time = current_time();
185  for (i = 0; i < NUM_BLOCKS; i++) {
186  ret = wc_AesCbcDecrypt(enc, bench_plain, bench_cipher, BLOCK_SIZE);
187  LWIP_ASSERT("ws_AesCbcDecrypt() failed", ret == 0);
188  }
189  done_time = current_time();
190  p = &result[19];
191  lwip_itoa(p, 5, SPEED(start_time, done_time));
192  while (*p)
193  p++;
194  *p = ' ';
195 
196  wc_AesFree(enc);
197 
198  return result;
199 }

222 /**************************************************************
223  * void close_socket(struct clientcb *p_clientcb)
224  *
225  * Close the socket and remove this clientcb from the list.
226  **************************************************************/
227 static void
228 close_socket(struct clientcb *p_clientcb)
229 {
230  struct clientcb *p_search_clientcb;
231 
232  /* Either an error or tcp connection closed on other
233  * end. Close here */
234  wolfSSL_free(p_clientcb->ssl);
235  close(p_clientcb->socket);
236  /* Free clientcb */
237  if (clientcb_list == p_clientcb) {
238  clientcb_list = p_clientcb->next;
239  } else {
240  for (p_search_clientcb = clientcb_list; p_search_clientcb; p_search_clientcb = p_search_clientcb->next) {
241  if (p_search_clientcb->next == p_clientcb) {
242  p_search_clientcb->next = p_clientcb->next;
243  break;
244  }
245  }
246  }
247  mem_free(p_clientcb);
248 }

262 /**************************************************************
263  * void sock_server_thread(void *arg)
264  *
265  * secure socket task. This server will wait for connections on
266  * TCP port number: PORT. For every connection, the server will
267  *echo back any message received.
268  **************************************************************/
269 static void
270 socket_server_thread(void *arg)
271 {
272  int listenfd;
273 #if LWIP_IPV6
274  struct sockaddr_in6 socket_saddr;
275 #else /* LWIP_IPV6 */
276  struct sockaddr_in socket_saddr;
277 #endif /* LWIP_IPV6 */
278  fd_set readset;
279  int i, maxfdp1;
280  int ret;
281  WOLFSSL_CTX* ctx;
282  struct clientcb *p_clientcb;
283  LWIP_UNUSED_ARG(arg);
284 
285  wait_for_ip();
286 
287  /* Initialize WOLFSSL */
288  ret = wolfSSL_Init();
289  LWIP_ASSERT("wolfSSL_Init() failed", ret == SSL_SUCCESS);
290 
291  memset(&socket_saddr, 0, sizeof (socket_saddr));
292 #if LWIP_IPV6
293  /* First acquire our socket for listening for connections */
294  listenfd = socket(AF_INET6, SOCK_STREAM, 0);
295  socket_saddr.sin6_family = AF_INET6;
296  socket_saddr.sin6_addr = in6addr_any;
297  socket_saddr.sin6_port = lwip_htons(PORT); /* echo server port */
298 #else /* LWIP_IPV6 */
299  /* First acquire our socket for listening for connections */
300  listenfd = socket(AF_INET, SOCK_STREAM, 0);
301  socket_saddr.sin_family = AF_INET;
302  socket_saddr.sin_addr.s_addr = PP_HTONL(INADDR_ANY);
303  socket_saddr.sin_port = lwip_htons(PORT); /* echo server port */
304 #endif /* LWIP_IPV6 */
305 
306  LWIP_ASSERT("socket_server_thread(): Socket create failed.", listenfd >= 0);
307 
308  /* Create and initialize CTX */
309  ctx = wolfSSL_CTX_new(wolfSSLv23_server_method());
310  LWIP_ASSERT("wolfSSL_CTX_new() failed", ctx != NULL);
311 
312  /* Limit to AES128 - hardware-accelerated */
313  wolfSSL_CTX_set_cipher_list(ctx, "AES128-SHA");
314 
315  /* Load CA certificates */
316  ret = wolfSSL_CTX_load_verify_buffer(ctx, CA_CERT, CA_CERT_SIZE, SSL_FILETYPE_ASN1);
317  LWIP_ASSERT("wolfSSL_CTX_load_verify_locations() failed", ret == SSL_SUCCESS);
318 
319  /* Load server certificate */
320  ret = wolfSSL_CTX_use_certificate_buffer(ctx, SERVER_CERT, SERVER_CERT_SIZE, SSL_FILETYPE_ASN1);
321  LWIP_ASSERT("wolfSSL_CTX_use_certificate_buffer() failed", ret == SSL_SUCCESS);
322 
323  /* Load keys */
324  ret = wolfSSL_CTX_use_PrivateKey_buffer(ctx, SERVER_KEY, SERVER_KEY_SIZE, SSL_FILETYPE_ASN1);
325  LWIP_ASSERT("wolfSSL_CTX_use_PrivateKey_buffer() failed", ret == SSL_SUCCESS);
326 
327  if (bind(listenfd, (struct sockaddr *) &socket_saddr, sizeof(socket_saddr)) == -1) {
328  LWIP_ASSERT("socket_server_thread(): Socket bind failed.", 0);
329  }
330 
331  /* Put socket into listening mode */
332  if (listen(listenfd, MAX_SERV) == -1) {
333  LWIP_ASSERT("socket_server_thread(): Listen failed.", 0);
334  }
335 
336  enc = (Aes*)mem_malloc(sizeof(Aes));
337  bench_plain = (byte*)mem_malloc(BLOCK_SIZE);
338  bench_cipher = (byte*)mem_malloc(BLOCK_SIZE);
339 
340  /* Wait forever for network input: This could be connections or data */
341  for (;;) {
342  maxfdp1 = listenfd + 1;
343 
344  /* Determine what sockets need to be in readset */
345  FD_ZERO(&readset);
346  FD_SET(listenfd, &readset);
347  for (p_clientcb = clientcb_list; p_clientcb; p_clientcb = p_clientcb->next) {
348  if (maxfdp1 < p_clientcb->socket + 1) {
349  maxfdp1 = p_clientcb->socket + 1;
350  }
351  FD_SET(p_clientcb->socket, &readset);
352  }
353 
354  /* Wait for data or a new connection */
355  i = select(maxfdp1, &readset, 0, 0, 0);
356 
357  if (i == 0) {
358  continue;
359  }
360  /* At least one descriptor is ready */
361  if (FD_ISSET(listenfd, &readset)) {
362  /* We have a new connection request!!! */
363  /* Lets create a new control block */
364  p_clientcb = (struct clientcb *) mem_malloc(sizeof (struct clientcb));
365  if (p_clientcb) {
366  p_clientcb->socket = accept(listenfd,
367  (struct sockaddr *) &p_clientcb->cliaddr,
368  &p_clientcb->clilen);
369  if (p_clientcb->socket < 0) {
370  mem_free(p_clientcb);
371  } else {
372  /* Keep this tecb in our list */
373  p_clientcb->ssl = wolfSSL_new(ctx);
374  LWIP_ASSERT("wolfSSL_new() failed.", p_clientcb->ssl != NULL);
375  wolfSSL_set_fd(p_clientcb->ssl, p_clientcb->socket);
376  p_clientcb->next = clientcb_list;
377  clientcb_list = p_clientcb;
378  }
379  } else {
380  /* No memory to accept connection. Just accept and then close */
381  int sock;
382  struct sockaddr cliaddr;
383  socklen_t clilen;
384 
385  sock = accept(listenfd, &cliaddr, &clilen);
386  if (sock >= 0) {
387  close(sock);
388  }
389  }
390  }
391  /* Go through list of connected clients and process data */
392  for (p_clientcb = clientcb_list; p_clientcb; p_clientcb = p_clientcb->next) {
393  if (FD_ISSET(p_clientcb->socket, &readset)) {
394  /* This socket is ready for reading. This could be because someone typed
395  * some characters or it could be because the socket is now closed. Try reading
396  * some data to see. */
397  int readcount;
398  readcount = wolfSSL_read(p_clientcb->ssl, &buffer, sizeof(buffer) - 1);
399  if (readcount <= 0) {
400  close_socket(p_clientcb);
401  break;
402  }
403  buffer[readcount] = 0;
404  if (!strncmp(buffer, "benchmark", 9)) {
405  strncpy(buffer, benchmark_crypto(), 40);
406  }
407  if (wolfSSL_write(p_clientcb->ssl, buffer, strlen(buffer)) < 0) {
408  close_socket(p_clientcb);
409  break;
410  }
411  }
412  }
413  }
414  mem_free(bench_plain);
415  mem_free(bench_cipher);
416  mem_free(enc);
417 
418  wolfSSL_CTX_free(ctx);
419  wolfSSL_Cleanup();
420  close(listenfd);
421 }
