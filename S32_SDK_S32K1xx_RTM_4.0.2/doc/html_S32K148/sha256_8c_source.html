381  #define Ch(x,y,z) ((z) ^ ((x) & ((y) ^ (z))))
382  #define Maj(x,y,z) ((((x) | (y)) & (z)) | ((x) & (y)))
383  #define R(x, n) (((x) & 0xFFFFFFFFU) >> (n))
384 
385  #define S(x, n) rotrFixed(x, n)
386  #define Sigma0(x) (S(x, 2) ^ S(x, 13) ^ S(x, 22))
387  #define Sigma1(x) (S(x, 6) ^ S(x, 11) ^ S(x, 25))
388  #define Gamma0(x) (S(x, 7) ^ S(x, 18) ^ R(x, 3))
389  #define Gamma1(x) (S(x, 17) ^ S(x, 19) ^ R(x, 10))

391  #define RND(a,b,c,d,e,f,g,h,i) \
392  t0 = (h) + Sigma1((e)) + Ch((e), (f), (g)) + K[(i)] + W[(i)]; \
393  t1 = Sigma0((a)) + Maj((a), (b), (c)); \
394  (d) += t0; \

401  static int Transform(Sha256* sha256)
402  {
403  word32 S[8], t0, t1;
404  int i;
405 
406  #ifdef WOLFSSL_SMALL_STACK
407  word32* W;
408 
409  W = (word32*)XMALLOC(sizeof(word32) * SHA256_BLOCK_SIZE, NULL,
410  DYNAMIC_TYPE_TMP_BUFFER);
411  if (W == NULL)
412  return MEMORY_E;
413  #else
414  word32 W[SHA256_BLOCK_SIZE];
415  #endif
416 
417  /* Copy context->state[] to working vars */
418  for (i = 0; i < 8; i++)
419  S[i] = sha256->digest[i];
420 
421  for (i = 0; i < 16; i++)
422  W[i] = sha256->buffer[i];
423 
424  for (i = 16; i < SHA256_BLOCK_SIZE; i++)
425  W[i] = Gamma1(W[i-2]) + W[i-7] + Gamma0(W[i-15]) + W[i-16];
426 
427  for (i = 0; i < SHA256_BLOCK_SIZE; i += 8) {
428  RND(S[0],S[1],S[2],S[3],S[4],S[5],S[6],S[7],i+0);
429  RND(S[7],S[0],S[1],S[2],S[3],S[4],S[5],S[6],i+1);
430  RND(S[6],S[7],S[0],S[1],S[2],S[3],S[4],S[5],i+2);
431  RND(S[5],S[6],S[7],S[0],S[1],S[2],S[3],S[4],i+3);
432  RND(S[4],S[5],S[6],S[7],S[0],S[1],S[2],S[3],i+4);
433  RND(S[3],S[4],S[5],S[6],S[7],S[0],S[1],S[2],i+5);
434  RND(S[2],S[3],S[4],S[5],S[6],S[7],S[0],S[1],i+6);
435  RND(S[1],S[2],S[3],S[4],S[5],S[6],S[7],S[0],i+7);
436  }
437 
438  /* Add the working vars back into digest state[] */
439  for (i = 0; i < 8; i++) {
440  sha256->digest[i] += S[i];
441  }
442 
443  #ifdef WOLFSSL_SMALL_STACK
444  XFREE(W, NULL, DYNAMIC_TYPE_TMP_BUFFER);
445  #endif
446 
447  return 0;
448  }
449 #endif

462  static INLINE int Sha256Update(Sha256* sha256, const byte* data, word32 len)
463  {
464  int ret = 0;
465  byte* local;
466 
467  if (sha256 == NULL || (data == NULL && len > 0)) {
468  return BAD_FUNC_ARG;
469  }
470 
471  #if defined(WOLFSSL_ASYNC_CRYPT) && defined(WC_ASYNC_ENABLE_SHA256)
472  if (sha256->asyncDev.marker == WOLFSSL_ASYNC_MARKER_SHA256) {
473  #if defined(HAVE_INTEL_QA)
474  return IntelQaSymSha256(&sha256->asyncDev, NULL, data, len);
475  #endif
476  }
477  #endif /* WOLFSSL_ASYNC_CRYPT */
478 
479  /* do block size increments */
480  local = (byte*)sha256->buffer;
481 
482  /* check that internal buffLen is valid */
483  if (sha256->buffLen >= SHA256_BLOCK_SIZE)
484  return BUFFER_E;
485 
486  SAVE_XMM_YMM; /* for Intel AVX */
487 
488  while (len) {
489  word32 add = min(len, SHA256_BLOCK_SIZE - sha256->buffLen);
490  XMEMCPY(&local[sha256->buffLen], data, add);
491 
492  sha256->buffLen += add;
493  data += add;
494  len -= add;
495 
496  if (sha256->buffLen == SHA256_BLOCK_SIZE) {
497  #if defined(LITTLE_ENDIAN_ORDER) && !defined(FREESCALE_MMCAU_SHA)
498  #if defined(HAVE_INTEL_AVX1) || defined(HAVE_INTEL_AVX2)
499  if (!IS_INTEL_AVX1(intel_flags) && !IS_INTEL_AVX2(intel_flags))
500  #endif
501  {
502  ByteReverseWords(sha256->buffer, sha256->buffer,
503  SHA256_BLOCK_SIZE);
504  }
505  #endif
506  ret = XTRANSFORM(sha256, local);
507  if (ret != 0) {
508  break;
509  }
510 
511  AddLength(sha256, SHA256_BLOCK_SIZE);
512  sha256->buffLen = 0;
513  }
514  }
515 
516  return ret;
517  }

524  static INLINE int Sha256Final(Sha256* sha256)
525  {
526 
527  int ret;
528  byte* local = (byte*)sha256->buffer;
529 
530  if (sha256 == NULL) {
531  return BAD_FUNC_ARG;
532  }
533 
534  SAVE_XMM_YMM; /* for Intel AVX */
535 
536  AddLength(sha256, sha256->buffLen); /* before adding pads */
537  local[sha256->buffLen++] = 0x80; /* add 1 */
538 
539  /* pad with zeros */
540  if (sha256->buffLen > SHA256_PAD_SIZE) {
541  XMEMSET(&local[sha256->buffLen], 0,
542  SHA256_BLOCK_SIZE - sha256->buffLen);
543  sha256->buffLen += SHA256_BLOCK_SIZE - sha256->buffLen;
544 
545  {
546  #if defined(LITTLE_ENDIAN_ORDER) && !defined(FREESCALE_MMCAU_SHA)
547  #if defined(HAVE_INTEL_AVX1) || defined(HAVE_INTEL_AVX2)
548  if (!IS_INTEL_AVX1(intel_flags) && !IS_INTEL_AVX2(intel_flags))
549  #endif
550  {
551  ByteReverseWords(sha256->buffer, sha256->buffer,
552  SHA256_BLOCK_SIZE);
553  }
554  #endif
555  }
556 
557  ret = XTRANSFORM(sha256, local);
558  if (ret != 0)
559  return ret;
560 
561  sha256->buffLen = 0;
562  }
563  XMEMSET(&local[sha256->buffLen], 0, SHA256_PAD_SIZE - sha256->buffLen);
564 
565  /* put lengths in bits */
566  sha256->hiLen = (sha256->loLen >> (8 * sizeof(sha256->loLen) - 3)) +
567  (sha256->hiLen << 3);
568  sha256->loLen = sha256->loLen << 3;
569 
570  /* store lengths */
571  #if defined(LITTLE_ENDIAN_ORDER) && !defined(FREESCALE_MMCAU_SHA)
572  #if defined(HAVE_INTEL_AVX1) || defined(HAVE_INTEL_AVX2)
573  if (!IS_INTEL_AVX1(intel_flags) && !IS_INTEL_AVX2(intel_flags))
574  #endif
575  {
576  ByteReverseWords(sha256->buffer, sha256->buffer,
577  SHA256_BLOCK_SIZE);
578  }
579  #endif
580  /* ! length ordering dependent on digest endian type ! */
581  XMEMCPY(&local[SHA256_PAD_SIZE], &sha256->hiLen, sizeof(word32));
582  XMEMCPY(&local[SHA256_PAD_SIZE + sizeof(word32)], &sha256->loLen,
583  sizeof(word32));
584 
585  #if defined(FREESCALE_MMCAU_SHA) || defined(HAVE_INTEL_AVX1) || \
586  defined(HAVE_INTEL_AVX2)
587  /* Kinetis requires only these bytes reversed */
588  #if defined(HAVE_INTEL_AVX1) || defined(HAVE_INTEL_AVX2)
589  if (IS_INTEL_AVX1(intel_flags) || IS_INTEL_AVX2(intel_flags))
590  #endif
591  {
592  ByteReverseWords(
593  &sha256->buffer[SHA256_PAD_SIZE / sizeof(word32)],
594  &sha256->buffer[SHA256_PAD_SIZE / sizeof(word32)],
595  2 * sizeof(word32));
596  }
597  #endif
598 
599  return XTRANSFORM(sha256, local);
600  }
