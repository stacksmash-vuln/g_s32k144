259 static status_t init(uint8_t phy)
260 {
261  status_t status;
262  static bool initialized;
263  phy_id_t id;
264  uint16_t config1;
265  uint8_t miiMode;
266 
267  if (initialized == false)
268  {
269  /* Make mapping to generic functions on first invocation of init function */
270  PHY_driver_tja110x.reset = PHY_driver_generic.reset;
271  PHY_driver_tja110x.getId = PHY_driver_generic.getId;
272  PHY_driver_tja110x.getLinkStatus = PHY_driver_generic.getLinkStatus;
273  PHY_driver_tja110x.getLinkSpeed = PHY_driver_generic.getLinkSpeed;
274  PHY_driver_tja110x.getDelay = PHY_driver_generic.getDelay;
275  PHY_driver_tja110x.suspend = PHY_driver_generic.suspend;
276  PHY_driver_tja110x.enableAutoneg = PHY_SHARED_ReturnNotSupported;
277  PHY_driver_tja110x.disableAutoneg = PHY_SHARED_ReturnNotSupported;
278  PHY_driver_tja110x.restartAutoneg = PHY_SHARED_ReturnNotSupported;
279 
280  initialized = true;
281  }
282 
283  status = PHY_driver_tja110x.getId(phy, &id);
284  if (status != STATUS_SUCCESS)
285  {
286  return status;
287  }
288 
289  status = ((id.oui != OUI) ? (STATUS_PHY_INCOMPATIBLE_DEVICE) : (STATUS_SUCCESS));
290  status = (((id.oui == 0U) && (id.typeNo == 0U) && (id.revisionNo == 0U)) ? (STATUS_SUCCESS) : (status)); /* If everything is 0, this is TJA1102_P1 */
291  if (status != STATUS_SUCCESS)
292  {
293  return status;
294  }
295 
296  if (id.typeNo == TJA1100_TYPE_NO)
297  {
298  g_phyConfig[phy].privateData |= PRIVATE_TJA1100;
299  }
300 
301 
302  status = handleInterrupt(phy); /* Handle any interrupts pending before initialization */
303  if (status != STATUS_SUCCESS)
304  {
305  return status;
306  }
307 
308  if (g_phyConfig[phy].manualDelay == false)
309  {
310  /* Read MII mode */
311  status = PHY_Read(phy, CONFIG1_ADDR, &config1);
312  if (status != STATUS_SUCCESS)
313  {
314  return status;
315  }
316 
317  miiMode = (uint8_t) ((config1 & CONFIG1_MII_MODE_MASK) >> CONFIG1_MII_MODE_SHIFT);
318 
319  if (miiMode == 0U)
320  {
321  /* MII mode */
322  g_phyConfig[phy].delay.miiToMdiMax = 300;
323  g_phyConfig[phy].delay.miiToMdiMin = 140;
324  g_phyConfig[phy].delay.mdiToMiiMax = 920;
325  g_phyConfig[phy].delay.mdiToMiiMin = 760;
326  }
327  else
328  {
329  /* RMII mode */
330  g_phyConfig[phy].delay.miiToMdiMax = 540;
331  g_phyConfig[phy].delay.miiToMdiMin = 190;
332  g_phyConfig[phy].delay.mdiToMiiMax = 1070;
333  g_phyConfig[phy].delay.mdiToMiiMin = 700;
334  }
335  }
336 
337  return STATUS_SUCCESS;
338 }

340 /*FUNCTION**********************************************************************
341  *
342  * Function Name : getState
343  * Description : Gets the current state of the specified PHY device
344  *
345  *END**************************************************************************/
346 static status_t getState(uint8_t phy, phy_state_t* state)
347 {
348  status_t status;
349  uint16_t extCtrl;
350 
351  status = PHY_Read(phy, EXTENDED_CTRL_ADDR, &extCtrl);
352  if (status != STATUS_SUCCESS)
353  {
354  return status;
355  }
356  extCtrl &= EXTENDED_CTRL_POWER_MODE_MASK;
357 
358  switch (extCtrl)
359  {
360  case EXTENDED_CTRL_POWER_MODE_SLEEP:
361  *state = PHY_STATE_SLEEP;
362  break;
363  case EXTENDED_CTRL_POWER_MODE_STANDBY:
364  *state = PHY_STATE_OFF;
365  break;
366  case EXTENDED_CTRL_POWER_MODE_SILENT:
367  *state = PHY_STATE_SILENT;
368  break;
369  case EXTENDED_CTRL_POWER_MODE_NORMAL_REQ:
370  *state = PHY_STATE_ACTIVE;
371  break;
372  case EXTENDED_CTRL_POWER_MODE_SLEEP_REQ:
373  *state = PHY_STATE_SLEEPREQ;
374  break;
375  default:
376  *state = PHY_STATE_UNKNOWN;
377  break;
378  }
379 
380  return STATUS_SUCCESS;
381 }

497 /*FUNCTION**********************************************************************
498  *
499  * Function Name : setLoopback
500  * Description : Sets the loopback mode for the specified PHY device.
501  *
502  *END**************************************************************************/
503 static status_t setLoopback(uint8_t phy, phy_loopback_t loopbackMode)
504 {
505  status_t status = STATUS_SUCCESS;
506  uint16_t extCtrl = 0;
507  uint16_t basicCtrl = BASIC_CTRL_LOOPBACK_FLAG;
508 
509  switch (loopbackMode)
510  {
511  case PHY_LOOPBACK_INTERNAL:
512  extCtrl = EXTENDED_CTRL_LOOPBACK_MODE_INTERNAL;
513  break;
514  case PHY_LOOPBACK_EXTERNAL:
515  extCtrl = EXTENDED_CTRL_LOOPBACK_MODE_EXTERNAL;
516  break;
517  case PHY_LOOPBACK_REMOTE:
518  extCtrl = EXTENDED_CTRL_LOOPBACK_MODE_REMOTE;
519  break;
520  case PHY_LOOPBACK_NONE:
521  extCtrl = 0;
522  basicCtrl = 0;
523  break;
524  default:
525  status = STATUS_UNSUPPORTED;
526  break;
527  }
528 
529  if (status == STATUS_SUCCESS)
530  {
531  if ((g_phyConfig[phy].privateData & PRIVATE_TJA1100) == 0U)
532  {
533  status = disableLink(phy);
534  if (status != STATUS_SUCCESS)
535  {
536  return status;
537  }
538  }
539  status = PHY_RMW(phy, EXTENDED_CTRL_ADDR, extCtrl, EXTENDED_CTRL_LOOPBACK_MODE_MASK);
540  if (status != STATUS_SUCCESS)
541  {
542  return status;
543  }
544 
545  status = PHY_RMW(phy, BASIC_CTRL_ADDR, basicCtrl, BASIC_CTRL_LOOPBACK_FLAG);
546  if (status != STATUS_SUCCESS)
547  {
548  return status;
549  }
550 
551  if ((g_phyConfig[phy].privateData & PRIVATE_TJA1100) == 0U)
552  {
553  status = enableLink(phy);
554  }
555  }
556 
557  return status;
558 }

615 /*FUNCTION**********************************************************************
616  *
617  * Function Name : handleInterrupt
618  * Description : Handles the interrupts for the specified PHY device.
619  *
620  *END**************************************************************************/
621 static status_t handleInterrupt(uint8_t phy)
622 {
623  status_t status;
624  uint16_t generalStatus;
625  uint16_t irqs;
626  bool pwonHandled = false;
627 
628  status = PHY_Read(phy, GENERAL_STATUS_ADDR, &generalStatus);
629  if (status != STATUS_SUCCESS)
630  {
631  return status;
632  }
633 
634  if ((generalStatus & (GENERAL_STATUS_RSTSTATUS_FLAG | GENERAL_STATUS_ENSTATUS_FLAG)) != 0U)
635  {
636  /* A resetting situation has occurred. Restore configuration */
637  pwonHandled = true;
638  status = handleInterruptPwon(phy);
639  if (status != STATUS_SUCCESS)
640  {
641  return status;
642  }
643  pwonHandled = true; /* pwon sequence already done. no need to do again */
644  }
645 
646  if ((generalStatus & GENERAL_STATUS_INT_STATUS_FLAG) != 0U)
647  {
648  status = PHY_Read(phy, IRQ_STATUS_ADDR, &irqs);
649  if (status != STATUS_SUCCESS)
650  {
651  return status;
652  }
653 
654  /* Interrupt handling */
655  if (((irqs & IRQ_PWON_FLAG) != 0U) && (pwonHandled == false))
656  {
657  status = handleInterruptPwon(phy);
658  if (status != STATUS_SUCCESS)
659  {
660  return status;
661  }
662  }
663 
664  if ((irqs & IRQ_WAKEUP_FLAG) != 0U)
665  {
666  status = handleInterruptWakeUp(phy);
667  if (status != STATUS_SUCCESS)
668  {
669  return status;
670  }
671  }
672 
673  if ((irqs & IRQ_LPS_RECEIVED_FLAG) != 0U)
674  {
675  handleInterruptSleep(phy);
676  }
677 
678  if ((irqs & IRQ_LINKDOWN_FLAG) != 0U)
679  {
680  handleInterruptLinkDown(phy);
681  }
682 
683  if ((irqs & IRQ_LINKUP_FLAG) != 0U)
684  {
685  status = handleInterruptLinkUp(phy);
686  if (status != STATUS_SUCCESS)
687  {
688  return status;
689  }
690  }
691  }
692 
693  return status;
694 }

702 static status_t writeExtendedConfig(uint8_t phy, bool config1, bool commonConfig)
703 {
704  status_t status = STATUS_SUCCESS;
705  uint16_t reg = 0;
706  uint16_t mask = 0;
707 
708  const phy_tja110x_config_t *extConfig = (phy_tja110x_config_t*) g_phyConfig[phy].extendedConfig;
709 
710  status = enableConfig(phy);
711  if (status != STATUS_SUCCESS)
712  {
713  return status;
714  }
715 
716  if (config1 == true)
717  {
718  /* configuration 1 */
719  reg |= (extConfig->fwdPhyLocal == true) ? (CONFIG1_FWDPHYLOC_FLAG) : (0U);
